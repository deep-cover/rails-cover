25 # Sets the default migration template that is being used for the generation of the migration.
26 # Depending on command line arguments, the migration template and the table name instance
27 # variables are set up.
28 def set_local_assigns!
29 @migration_template = "migration.rb"
30 case file_name
31 when /^(add)_.*_to_(.*)/, /^(remove)_.*?_from_(.*)/
32 @migration_action = $1
33 @table_name = normalize_table_name($2)
34 when /join_table/
35 if attributes.length == 2
36 @migration_action = "join"
37 @join_tables = pluralize_table_names? ? attributes.map(&:plural_name) : attributes.map(&:singular_name)
38
39 set_index_names
40 end
41 when /^create_(.+)/
42 @table_name = normalize_table_name($1)
43 @migration_template = "create_table_migration.rb"
44 end
45 end
46
47 def set_index_names
48 attributes.each_with_index do |attr, i|
49 attr.index_name = [attr, attributes[i - 1]].map { |a| index_name_for(a) }
50 end
51 end
52
53 def index_name_for(attribute)
54 if attribute.foreign_key?
55 attribute.name
56 else
57 attribute.name.singularize.foreign_key
58 end.to_sym
59 end
