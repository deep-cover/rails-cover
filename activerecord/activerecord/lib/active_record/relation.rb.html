106 # Finds the first record with the given attributes, or creates a record
107 # with the attributes if one is not found:
108 #
109 # # Find the first user named "Penélope" or create a new one.
110 # User.find_or_create_by(first_name: 'Penélope')
111 # # => #<User id: 1, first_name: "Penélope", last_name: nil>
112 #
113 # # Find the first user named "Penélope" or create a new one.
114 # # We already have one so the existing record will be returned.
115 # User.find_or_create_by(first_name: 'Penélope')
116 # # => #<User id: 1, first_name: "Penélope", last_name: nil>
117 #
118 # # Find the first user named "Scarlett" or create a new one with
119 # # a particular last name.
120 # User.create_with(last_name: 'Johansson').find_or_create_by(first_name: 'Scarlett')
121 # # => #<User id: 2, first_name: "Scarlett", last_name: "Johansson">
122 #
123 # This method accepts a block, which is passed down to #create. The last example
124 # above can be alternatively written this way:
125 #
126 # # Find the first user named "Scarlett" or create a new one with a
127 # # different last name.
128 # User.find_or_create_by(first_name: 'Scarlett') do |user|
129 # user.last_name = 'Johansson'
130 # end
131 # # => #<User id: 2, first_name: "Scarlett", last_name: "Johansson">
132 #
133 # This method always returns a record, but if creation was attempted and
134 # failed due to validation errors it won't be persisted, you get what
135 # #create returns in such situation.
136 #
137 # Please note *this method is not atomic*, it runs first a SELECT, and if
138 # there are no results an INSERT is attempted. If there are other threads
139 # or processes there is a race condition between both calls and it could
140 # be the case that you end up with two similar records.
141 #
142 # Whether that is a problem or not depends on the logic of the
143 # application, but in the particular case in which rows have a UNIQUE
144 # constraint an exception may be raised, just retry:
145 #
146 # begin
147 # CreditAccount.transaction(requires_new: true) do
148 # CreditAccount.find_or_create_by(user_id: user.id)
149 # end
150 # rescue ActiveRecord::RecordNotUnique
151 # retry
152 # end
153 #
154 def find_or_create_by(attributes, &block)
155 find_by(attributes) || create(attributes, &block)
156 end

235 # Returns a cache key that can be used to identify the records fetched by
236 # this query. The cache key is built with a fingerprint of the sql query,
237 # the number of records matched by the query and a timestamp of the last
238 # updated record. When a new record comes to match the query, or any of
239 # the existing records is updated or deleted, the cache key changes.
240 #
241 # Product.where("name like ?", "%Cosmic Encounter%").cache_key
242 # # => "products/query-1850ab3d302391b85b8693e941286659-1-20150714212553907087000"
243 #
244 # If the collection is loaded, the method will iterate through the records
245 # to generate the timestamp, otherwise it will trigger one SQL query like:
246 #
247 # SELECT COUNT(*), MAX("products"."updated_at") FROM "products" WHERE (name like '%Cosmic Encounter%')
248 #
249 # You can also pass a custom timestamp column to fetch the timestamp of the
250 # last updated record.
251 #
252 # Product.where("name like ?", "%Game%").cache_key(:last_reviewed_at)
253 #
254 # You can customize the strategy to generate the key on a per model basis
255 # overriding ActiveRecord::Base#collection_cache_key.
256 def cache_key(timestamp_column = :updated_at)
257 @cache_keys ||= {}
258 @cache_keys[timestamp_column] ||= @klass.collection_cache_key(self, timestamp_column)
259 end

277 # Updates all records in the current relation with details given. This method constructs a single SQL UPDATE
278 # statement and sends it straight to the database. It does not instantiate the involved models and it does not
279 # trigger Active Record callbacks or validations. However, values passed to #update_all will still go through
280 # Active Record's normal type casting and serialization.
281 #
282 # ==== Parameters
283 #
284 # * +updates+ - A string, array, or hash representing the SET part of an SQL statement.
285 #
286 # ==== Examples
287 #
288 # # Update all customers with the given attributes
289 # Customer.update_all wants_email: true
290 #
291 # # Update all books with 'Rails' in their title
292 # Book.where('title LIKE ?', '%Rails%').update_all(author: 'David')
293 #
294 # # Update all books that match conditions, but limit it to 5 ordered by date
295 # Book.where('title LIKE ?', '%Rails%').order(:created_at).limit(5).update_all(author: 'David')
296 #
297 # # Update all invoices and set the number column to its id value.
298 # Invoice.update_all('number = id')
299 def update_all(updates)
300 raise ArgumentError, "Empty list of attributes to change" if updates.blank?
301
302 if eager_loading?
303 relation = apply_join_dependency
304 return relation.update_all(updates)
305 end
306
307 stmt = Arel::UpdateManager.new
308
309 stmt.set Arel.sql(@klass.send(:sanitize_sql_for_assignment, updates))
310 stmt.table(table)
311
312 if has_join_values?
313 @klass.connection.join_to_update(stmt, arel, arel_attribute(primary_key))
314 else
315 stmt.key = arel_attribute(primary_key)
316 stmt.take(arel.limit)
317 stmt.order(*arel.orders)
318 stmt.wheres = arel.constraints
319 end
320
321 @klass.connection.update stmt, "#{@klass} Update All"
322 end

324 # Destroys the records by instantiating each
325 # record and calling its {#destroy}[rdoc-ref:Persistence#destroy] method.
326 # Each object's callbacks are executed (including <tt>:dependent</tt> association options).
327 # Returns the collection of objects that were destroyed; each will be frozen, to
328 # reflect that no changes should be made (since they can't be persisted).
329 #
330 # Note: Instantiation, callback execution, and deletion of each
331 # record can be time consuming when you're removing many records at
332 # once. It generates at least one SQL +DELETE+ query per record (or
333 # possibly more, to enforce your callbacks). If you want to delete many
334 # rows quickly, without concern for their associations or callbacks, use
335 # #delete_all instead.
336 #
337 # ==== Examples
338 #
339 # Person.where(age: 0..18).destroy_all
340 def destroy_all
341 records.each(&:destroy).tap { reset }
342 end

344 # Deletes the records without instantiating the records
345 # first, and hence not calling the {#destroy}[rdoc-ref:Persistence#destroy]
346 # method nor invoking callbacks.
347 # This is a single SQL DELETE statement that goes straight to the database, much more
348 # efficient than #destroy_all. Be careful with relations though, in particular
349 # <tt>:dependent</tt> rules defined on associations are not honored. Returns the
350 # number of rows affected.
351 #
352 # Post.where(person_id: 5).where(category: ['Something', 'Else']).delete_all
353 #
354 # Both calls delete the affected posts all at once with a single DELETE statement.
355 # If you need to destroy dependent associations or call your <tt>before_*</tt> or
356 # +after_destroy+ callbacks, use the #destroy_all method instead.
357 #
358 # If an invalid method is supplied, #delete_all raises an ActiveRecordError:
359 #
360 # Post.limit(100).delete_all
361 # # => ActiveRecord::ActiveRecordError: delete_all doesn't support limit
362 def delete_all
363 invalid_methods = INVALID_METHODS_FOR_DELETE_ALL.select do |method|
364 value = get_value(method)
365 SINGLE_VALUE_METHODS.include?(method) ? value : value.any?
366 end
367 if invalid_methods.any?
368 raise ActiveRecordError.new("delete_all doesn't support #{invalid_methods.join(', ')}")
369 end
370
371 if eager_loading?
372 relation = apply_join_dependency
373 return relation.delete_all
374 end
375
376 stmt = Arel::DeleteManager.new
377 stmt.from(table)
378
379 if has_join_values?
380 @klass.connection.join_to_delete(stmt, arel, arel_attribute(primary_key))
381 else
382 stmt.wheres = arel.constraints
383 end
384
385 affected = @klass.connection.delete(stmt, "#{@klass} Destroy")
386
387 reset
388 affected
389 end

522 def exec_queries(&block)
523 skip_query_cache_if_necessary do
524 @records =
525 if eager_loading?
526 find_with_associations do |relation, join_dependency|
527 if ActiveRecord::NullRelation === relation
528 []
529 else
530 rows = connection.select_all(relation.arel, "SQL")
531 join_dependency.instantiate(rows, &block)
532 end.freeze
533 end
534 else
535 klass.find_by_sql(arel, &block).freeze
536 end

538 preload = preload_values
539 preload += includes_values unless eager_loading?
540 preloader = nil
541 preload.each do |associations|
542 preloader ||= build_preloader
543 preloader.preload @records, associations
544 end
545
546 @records.each(&:readonly!) if readonly_value
547
548 @loaded = true
549 @records
550 end
