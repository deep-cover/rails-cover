93 def get_primary_key(base_name) #:nodoc:
94 if base_name && primary_key_prefix_type == :table_name
95 base_name.foreign_key(false)
96 elsif base_name && primary_key_prefix_type == :table_name_with_underscore
97 base_name.foreign_key
98 else
99 if ActiveRecord::Base != self && table_exists?
100 pk = connection.schema_cache.primary_keys(table_name)

103 "id"
