183 # = Active Record Migrations
184 #
185 # Migrations can manage the evolution of a schema used by several physical
186 # databases. It's a solution to the common problem of adding a field to make
187 # a new feature work in your local database, but being unsure of how to
188 # push that change to other developers and to the production server. With
189 # migrations, you can describe the transformations in self-contained classes
190 # that can be checked into version control systems and executed against
191 # another database that might be one, two, or five versions behind.
192 #
193 # Example of a simple migration:
194 #
195 # class AddSsl < ActiveRecord::Migration[5.0]
196 # def up
197 # add_column :accounts, :ssl_enabled, :boolean, default: true
198 # end
199 #
200 # def down
201 # remove_column :accounts, :ssl_enabled
202 # end
203 # end
204 #
205 # This migration will add a boolean flag to the accounts table and remove it
206 # if you're backing out of the migration. It shows how all migrations have
207 # two methods +up+ and +down+ that describes the transformations
208 # required to implement or remove the migration. These methods can consist
209 # of both the migration specific methods like +add_column+ and +remove_column+,
210 # but may also contain regular Ruby code for generating data needed for the
211 # transformations.
212 #
213 # Example of a more complex migration that also needs to initialize data:
214 #
215 # class AddSystemSettings < ActiveRecord::Migration[5.0]
216 # def up
217 # create_table :system_settings do |t|
218 # t.string :name
219 # t.string :label
220 # t.text :value
221 # t.string :type
222 # t.integer :position
223 # end
224 #
225 # SystemSetting.create name: 'notice',
226 # label: 'Use notice?',
227 # value: 1
228 # end
229 #
230 # def down
231 # drop_table :system_settings
232 # end
233 # end
234 #
235 # This migration first adds the +system_settings+ table, then creates the very
236 # first row in it using the Active Record model that relies on the table. It
237 # also uses the more advanced +create_table+ syntax where you can specify a
238 # complete table schema in one block call.
239 #
240 # == Available transformations
241 #
242 # === Creation
243 #
244 # * <tt>create_join_table(table_1, table_2, options)</tt>: Creates a join
245 # table having its name as the lexical order of the first two
246 # arguments. See
247 # ActiveRecord::ConnectionAdapters::SchemaStatements#create_join_table for
248 # details.
249 # * <tt>create_table(name, options)</tt>: Creates a table called +name+ and
250 # makes the table object available to a block that can then add columns to it,
251 # following the same format as +add_column+. See example above. The options hash
252 # is for fragments like "DEFAULT CHARSET=UTF-8" that are appended to the create
253 # table definition.
254 # * <tt>add_column(table_name, column_name, type, options)</tt>: Adds a new column
255 # to the table called +table_name+
256 # named +column_name+ specified to be one of the following types:
257 # <tt>:string</tt>, <tt>:text</tt>, <tt>:integer</tt>, <tt>:float</tt>,
258 # <tt>:decimal</tt>, <tt>:datetime</tt>, <tt>:timestamp</tt>, <tt>:time</tt>,
259 # <tt>:date</tt>, <tt>:binary</tt>, <tt>:boolean</tt>. A default value can be
260 # specified by passing an +options+ hash like <tt>{ default: 11 }</tt>.
261 # Other options include <tt>:limit</tt> and <tt>:null</tt> (e.g.
262 # <tt>{ limit: 50, null: false }</tt>) -- see
263 # ActiveRecord::ConnectionAdapters::TableDefinition#column for details.
264 # * <tt>add_foreign_key(from_table, to_table, options)</tt>: Adds a new
265 # foreign key. +from_table+ is the table with the key column, +to_table+ contains
266 # the referenced primary key.
267 # * <tt>add_index(table_name, column_names, options)</tt>: Adds a new index
268 # with the name of the column. Other options include
269 # <tt>:name</tt>, <tt>:unique</tt> (e.g.
270 # <tt>{ name: 'users_name_index', unique: true }</tt>) and <tt>:order</tt>
271 # (e.g. <tt>{ order: { name: :desc } }</tt>).
272 # * <tt>add_reference(:table_name, :reference_name)</tt>: Adds a new column
273 # +reference_name_id+ by default an integer. See
274 # ActiveRecord::ConnectionAdapters::SchemaStatements#add_reference for details.
275 # * <tt>add_timestamps(table_name, options)</tt>: Adds timestamps (+created_at+
276 # and +updated_at+) columns to +table_name+.
277 #
278 # === Modification
279 #
280 # * <tt>change_column(table_name, column_name, type, options)</tt>: Changes
281 # the column to a different type using the same parameters as add_column.
282 # * <tt>change_column_default(table_name, column_name, default_or_changes)</tt>:
283 # Sets a default value for +column_name+ defined by +default_or_changes+ on
284 # +table_name+. Passing a hash containing <tt>:from</tt> and <tt>:to</tt>
285 # as +default_or_changes+ will make this change reversible in the migration.
286 # * <tt>change_column_null(table_name, column_name, null, default = nil)</tt>:
287 # Sets or removes a +NOT NULL+ constraint on +column_name+. The +null+ flag
288 # indicates whether the value can be +NULL+. See
289 # ActiveRecord::ConnectionAdapters::SchemaStatements#change_column_null for
290 # details.
291 # * <tt>change_table(name, options)</tt>: Allows to make column alterations to
292 # the table called +name+. It makes the table object available to a block that
293 # can then add/remove columns, indexes or foreign keys to it.
294 # * <tt>rename_column(table_name, column_name, new_column_name)</tt>: Renames
295 # a column but keeps the type and content.
296 # * <tt>rename_index(table_name, old_name, new_name)</tt>: Renames an index.
297 # * <tt>rename_table(old_name, new_name)</tt>: Renames the table called +old_name+
298 # to +new_name+.
299 #
300 # === Deletion
301 #
302 # * <tt>drop_table(name)</tt>: Drops the table called +name+.
303 # * <tt>drop_join_table(table_1, table_2, options)</tt>: Drops the join table
304 # specified by the given arguments.
305 # * <tt>remove_column(table_name, column_name, type, options)</tt>: Removes the column
306 # named +column_name+ from the table called +table_name+.
307 # * <tt>remove_columns(table_name, *column_names)</tt>: Removes the given
308 # columns from the table definition.
309 # * <tt>remove_foreign_key(from_table, options_or_to_table)</tt>: Removes the
310 # given foreign key from the table called +table_name+.
311 # * <tt>remove_index(table_name, column: column_names)</tt>: Removes the index
312 # specified by +column_names+.
313 # * <tt>remove_index(table_name, name: index_name)</tt>: Removes the index
314 # specified by +index_name+.
315 # * <tt>remove_reference(table_name, ref_name, options)</tt>: Removes the
316 # reference(s) on +table_name+ specified by +ref_name+.
317 # * <tt>remove_timestamps(table_name, options)</tt>: Removes the timestamp
318 # columns (+created_at+ and +updated_at+) from the table definition.
319 #
320 # == Irreversible transformations
321 #
322 # Some transformations are destructive in a manner that cannot be reversed.
323 # Migrations of that kind should raise an <tt>ActiveRecord::IrreversibleMigration</tt>
324 # exception in their +down+ method.
325 #
326 # == Running migrations from within Rails
327 #
328 # The Rails package has several tools to help create and apply migrations.
329 #
330 # To generate a new migration, you can use
331 # rails generate migration MyNewMigration
332 #
333 # where MyNewMigration is the name of your migration. The generator will
334 # create an empty migration file <tt>timestamp_my_new_migration.rb</tt>
335 # in the <tt>db/migrate/</tt> directory where <tt>timestamp</tt> is the
336 # UTC formatted date and time that the migration was generated.
337 #
338 # There is a special syntactic shortcut to generate migrations that add fields to a table.
339 #
340 # rails generate migration add_fieldname_to_tablename fieldname:string
341 #
342 # This will generate the file <tt>timestamp_add_fieldname_to_tablename.rb</tt>, which will look like this:
343 # class AddFieldnameToTablename < ActiveRecord::Migration[5.0]
344 # def change
345 # add_column :tablenames, :fieldname, :string
346 # end
347 # end
348 #
349 # To run migrations against the currently configured database, use
350 # <tt>rails db:migrate</tt>. This will update the database by running all of the
351 # pending migrations, creating the <tt>schema_migrations</tt> table
352 # (see "About the schema_migrations table" section below) if missing. It will also
353 # invoke the db:schema:dump task, which will update your db/schema.rb file
354 # to match the structure of your database.
355 #
356 # To roll the database back to a previous migration version, use
357 # <tt>rails db:rollback VERSION=X</tt> where <tt>X</tt> is the version to which
358 # you wish to downgrade. Alternatively, you can also use the STEP option if you
359 # wish to rollback last few migrations. <tt>rails db:rollback STEP=2</tt> will rollback
360 # the latest two migrations.
361 #
362 # If any of the migrations throw an <tt>ActiveRecord::IrreversibleMigration</tt> exception,
363 # that step will fail and you'll have some manual work to do.
364 #
365 # == Database support
366 #
367 # Migrations are currently supported in MySQL, PostgreSQL, SQLite,
368 # SQL Server, and Oracle (all supported databases except DB2).
369 #
370 # == More examples
371 #
372 # Not all migrations change the schema. Some just fix the data:
373 #
374 # class RemoveEmptyTags < ActiveRecord::Migration[5.0]
375 # def up
376 # Tag.all.each { |tag| tag.destroy if tag.pages.empty? }
377 # end
378 #
379 # def down
380 # # not much we can do to restore deleted data
381 # raise ActiveRecord::IrreversibleMigration, "Can't recover the deleted tags"
382 # end
383 # end
384 #
385 # Others remove columns when they migrate up instead of down:
386 #
387 # class RemoveUnnecessaryItemAttributes < ActiveRecord::Migration[5.0]
388 # def up
389 # remove_column :items, :incomplete_items_count
390 # remove_column :items, :completed_items_count
391 # end
392 #
393 # def down
394 # add_column :items, :incomplete_items_count
395 # add_column :items, :completed_items_count
396 # end
397 # end
398 #
399 # And sometimes you need to do something in SQL not abstracted directly by migrations:
400 #
401 # class MakeJoinUnique < ActiveRecord::Migration[5.0]
402 # def up
403 # execute "ALTER TABLE `pages_linked_pages` ADD UNIQUE `page_id_linked_page_id` (`page_id`,`linked_page_id`)"
404 # end
405 #
406 # def down
407 # execute "ALTER TABLE `pages_linked_pages` DROP INDEX `page_id_linked_page_id`"
408 # end
409 # end

411 # == Using a model after changing its table
412 #
413 # Sometimes you'll want to add a column in a migration and populate it
414 # immediately after. In that case, you'll need to make a call to
415 # <tt>Base#reset_column_information</tt> in order to ensure that the model has the
416 # latest column data from after the new column was added. Example:
417 #
418 # class AddPeopleSalary < ActiveRecord::Migration[5.0]
419 # def up
420 # add_column :people, :salary, :integer
421 # Person.reset_column_information
422 # Person.all.each do |p|
423 # p.update_attribute :salary, SalaryCalculator.compute(p)
424 # end
425 # end
426 # end
427 #
428 # == Controlling verbosity
429 #
430 # By default, migrations will describe the actions they are taking, writing
431 # them to the console as they happen, along with benchmarks describing how
432 # long each step took.
433 #
434 # You can quiet them down by setting ActiveRecord::Migration.verbose = false.
435 #
436 # You can also insert your own messages and benchmarks by using the +say_with_time+
437 # method:
438 #
439 # def up
440 # ...
441 # say_with_time "Updating salaries..." do
442 # Person.all.each do |p|
443 # p.update_attribute :salary, SalaryCalculator.compute(p)
444 # end
445 # end
446 # ...
447 # end
448 #
449 # The phrase "Updating salaries..." would then be printed, along with the
450 # benchmark for the block when the block completes.
451 #
452 # == Timestamped Migrations
453 #
454 # By default, Rails generates migrations that look like:
455 #
456 # 20080717013526_your_migration_name.rb
457 #
458 # The prefix is a generation timestamp (in UTC).
459 #
460 # If you'd prefer to use numeric prefixes, you can turn timestamped migrations
461 # off by setting:
462 #
463 # config.active_record.timestamped_migrations = false
464 #
465 # In application.rb.
466 #
467 # == Reversible Migrations
468 #
469 # Reversible migrations are migrations that know how to go +down+ for you.
470 # You simply supply the +up+ logic, and the Migration system figures out
471 # how to execute the down commands for you.
472 #
473 # To define a reversible migration, define the +change+ method in your
474 # migration like this:
475 #
476 # class TenderloveMigration < ActiveRecord::Migration[5.0]
477 # def change
478 # create_table(:horses) do |t|
479 # t.column :content, :text
480 # t.column :remind_at, :datetime
481 # end
482 # end
483 # end
484 #
485 # This migration will create the horses table for you on the way up, and
486 # automatically figure out how to drop the table on the way down.
487 #
488 # Some commands like +remove_column+ cannot be reversed. If you care to
489 # define how to move up and down in these cases, you should define the +up+
490 # and +down+ methods as before.
491 #
492 # If a command cannot be reversed, an
493 # <tt>ActiveRecord::IrreversibleMigration</tt> exception will be raised when
494 # the migration is moving down.
495 #
496 # For a list of commands that are reversible, please see
497 # <tt>ActiveRecord::Migration::CommandRecorder</tt>.
498 #
499 # == Transactional Migrations
500 #
501 # If the database adapter supports DDL transactions, all migrations will
502 # automatically be wrapped in a transaction. There are queries that you
503 # can't execute inside a transaction though, and for these situations
504 # you can turn the automatic transactions off.
505 #
506 # class ChangeEnum < ActiveRecord::Migration[5.0]
507 # disable_ddl_transaction!
508 #
509 # def up
510 # execute "ALTER TYPE model_size ADD VALUE 'new_value'"
511 # end
512 # end
513 #
514 # Remember that you can still open your own transactions, even if you
515 # are in a Migration with <tt>self.disable_ddl_transaction!</tt>.

636 # Reverses the migration commands for the given block and
637 # the given migrations.
638 #
639 # The following migration will remove the table 'horses'
640 # and create the table 'apples' on the way up, and the reverse
641 # on the way down.
642 #
643 # class FixTLMigration < ActiveRecord::Migration[5.0]
644 # def change
645 # revert do
646 # create_table(:horses) do |t|
647 # t.text :content
648 # t.datetime :remind_at
649 # end
650 # end
651 # create_table(:apples) do |t|
652 # t.string :variety
653 # end
654 # end
655 # end
656 #
657 # Or equivalently, if +TenderloveMigration+ is defined as in the
658 # documentation for Migration:
659 #
660 # require_relative '20121212123456_tenderlove_migration'
661 #
662 # class FixupTLMigration < ActiveRecord::Migration[5.0]
663 # def change
664 # revert TenderloveMigration
665 #
666 # create_table(:apples) do |t|
667 # t.string :variety
668 # end
669 # end
670 # end
671 #
672 # This command can be nested.
673 def revert(*migration_classes)
674 run(*migration_classes.reverse, revert: true) unless migration_classes.empty?
675 if block_given?
676 if connection.respond_to? :revert
677 connection.revert { yield }
678 else
679 recorder = CommandRecorder.new(connection)
680 @connection = recorder
681 suppress_messages do
682 connection.revert { yield }
683 end
684 @connection = recorder.delegate
685 recorder.commands.each do |cmd, args, block|
686 send(cmd, *args, &block)
687 end
688 end
689 end
690 end

707 # Call the methods +up+ and +down+ of the yielded object to run a block
708 # only in one given direction.
709 # The whole block will be called in the right order within the migration.
710 #
711 # In the following example, the looping on users will always be done
712 # when the three columns 'first_name', 'last_name' and 'full_name' exist,
713 # even when migrating down:
714 #
715 # class SplitNameMigration < ActiveRecord::Migration[5.0]
716 # def change
717 # add_column :users, :first_name, :string
718 # add_column :users, :last_name, :string
719 #
720 # reversible do |dir|
721 # User.reset_column_information
722 # User.all.each do |u|
723 # dir.up { u.first_name, u.last_name = u.full_name.split(' ') }
724 # dir.down { u.full_name = "#{u.first_name} #{u.last_name}" }
725 # u.save
726 # end
727 # end
728 #
729 # revert { add_column :users, :full_name, :string }
730 # end
731 # end
732 def reversible
733 helper = ReversibleBlockHelper.new(reverting?)
734 execute_block { yield helper }
735 end

737 # Used to specify an operation that is only run when migrating up
738 # (for example, populating a new column with its initial values).
739 #
740 # In the following example, the new column +published+ will be given
741 # the value +true+ for all existing records.
742 #
743 # class AddPublishedToPosts < ActiveRecord::Migration[5.2]
744 # def change
745 # add_column :posts, :published, :boolean, default: false
746 # up_only do
747 # execute "update posts set published = 'true'"
748 # end
749 # end
750 # end
751 def up_only
752 execute_block { yield } unless reverting?
753 end

755 # Runs the given migration classes.
756 # Last argument can specify options:
757 # - :direction (default is :up)
758 # - :revert (default is false)
759 def run(*migration_classes)
760 opts = migration_classes.extract_options!
761 dir = opts[:direction] || :up
762 dir = (dir == :down ? :up : :down) if opts[:revert]
763 if reverting?
764 # If in revert and going :up, say, we want to execute :down without reverting, so
765 revert { run(*migration_classes, direction: dir, revert: true) }
766 else
767 migration_classes.each do |migration_class|
768 migration_class.new.exec_migration(connection, dir)
769 end
770 end
771 end

874 def copy(destination, sources, options = {})
875 copied = []
876
877 FileUtils.mkdir_p(destination) unless File.exist?(destination)
878
879 destination_migrations = ActiveRecord::Migrator.migrations(destination)
880 last = destination_migrations.last
881 sources.each do |scope, path|
882 source_migrations = ActiveRecord::Migrator.migrations(path)
883
884 source_migrations.each do |migration|
885 source = File.binread(migration.filename)
886 inserted_comment = "# This migration comes from #{scope} (originally #{migration.version})\n"
887 magic_comments = "".dup
888 loop do
889 # If we have a magic comment in the original migration,
890 # insert our comment after the first newline(end of the magic comment line)
891 # so the magic keep working.
892 # Note that magic comments must be at the first line(except sh-bang).
893 source.sub!(/\A(?:#.*\b(?:en)?coding:\s*\S+|#\s*frozen_string_literal:\s*(?:true|false)).*\n/) do |magic_comment|
894 magic_comments << magic_comment; ""
895 end || break
896 end
897 source = "#{magic_comments}#{inserted_comment}#{source}"
898
899 if duplicate = destination_migrations.detect { |m| m.name == migration.name }
900 if options[:on_skip] && duplicate.scope != scope.to_s
901 options[:on_skip].call(scope, migration)
902 end
903 next
904 end
905
906 migration.version = next_migration_number(last ? last.version + 1 : 0).to_i
907 new_path = File.join(destination, "#{migration.version}_#{migration.name.underscore}.#{scope}.rb")
908 old_path, migration.filename = migration.filename, new_path
909 last = migration
910
911 File.binwrite(migration.filename, source)
912 copied << migration
913 options[:on_copy].call(scope, migration, old_path) if options[:on_copy]
914 destination_migrations << migration
915 end
916 end
917
918 copied
919 end

1251 return if down? && !migrated.include?(migration.version.to_i)
1252 return if up? && migrated.include?(migration.version.to_i)
