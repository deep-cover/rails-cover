5module ActiveRecord
6 # == Single table inheritance
7 #
8 # Active Record allows inheritance by storing the name of the class in a column that by
9 # default is named "type" (can be changed by overwriting <tt>Base.inheritance_column</tt>).
10 # This means that an inheritance looking like this:
11 #
12 # class Company < ActiveRecord::Base; end
13 # class Firm < Company; end
14 # class Client < Company; end
15 # class PriorityClient < Client; end
16 #
17 # When you do <tt>Firm.create(name: "37signals")</tt>, this record will be saved in
18 # the companies table with type = "Firm". You can then fetch this row again using
19 # <tt>Company.where(name: '37signals').first</tt> and it will return a Firm object.
20 #
21 # Be aware that because the type column is an attribute on the record every new
22 # subclass will instantly be marked as dirty and the type column will be included
23 # in the list of changed attributes on the record. This is different from non
24 # Single Table Inheritance(STI) classes:
25 #
26 # Company.new.changed? # => false
27 # Firm.new.changed? # => true
28 # Firm.new.changes # => {"type"=>["","Firm"]}
29 #
30 # If you don't have a type column defined in your table, single-table inheritance won't
31 # be triggered. In that case, it'll work just like normal subclasses with no special magic
32 # for differentiating between them or reloading the right type with find.
33 #
34 # Note, all the attributes for all the cases are kept in the same table. Read more:
35 # https://www.martinfowler.com/eaaCatalog/singleTableInheritance.html
36 #
37 module Inheritance

88 # Returns the class descending directly from ActiveRecord::Base, or
89 # an abstract class, if any, in the inheritance hierarchy.
90 #
91 # If A extends ActiveRecord::Base, A.base_class will return A. If B descends from A
92 # through some arbitrarily deep hierarchy, B.base_class will return A.
93 #
94 # If B < A and C < B and if A is an abstract_class then both B.base_class
95 # and C.base_class would return B as the answer since A is an abstract_class.
96 def base_class
97 unless self < Base
98 raise ActiveRecordError, "#{name} doesn't belong in a hierarchy descending from ActiveRecord"
99 end
100
101 if superclass == Base || superclass.abstract_class?
102 self
103 else
104 superclass.base_class
105 end
106 end

108 # Set this to true if this is an abstract class (see <tt>abstract_class?</tt>).
109 # If you are using inheritance with ActiveRecord and don't want child classes
110 # to utilize the implied STI table name of the parent class, this will need to be true.
111 # For example, given the following:
112 #
113 # class SuperClass < ActiveRecord::Base
114 # self.abstract_class = true
115 # end
116 # class Child < SuperClass
117 # self.table_name = 'the_table_i_really_want'
118 # end
119 #
120 #
121 # <tt>self.abstract_class = true</tt> is required to make <tt>Child<.find,.create, or any Arel method></tt> use <tt>the_table_i_really_want</tt> instead of a table called <tt>super_classes</tt>
122 #

141 # Returns the class type of the record using the current module as a prefix. So descendants of
142 # MyApp::Business::Account would appear as MyApp::Business::AccountSubclass.
143 def compute_type(type_name)
144 if type_name.start_with?("::".freeze)
145 # If the type is prefixed with a scope operator then we assume that
146 # the type_name is an absolute reference.
147 ActiveSupport::Dependencies.constantize(type_name)
148 else
149 type_candidate = @_type_candidates_cache[type_name]
150 if type_candidate && type_constant = ActiveSupport::Dependencies.safe_constantize(type_candidate)
151 return type_constant
152 end
153
154 # Build a list of candidates to search for
155 candidates = []
156 name.scan(/::|$/) { candidates.unshift "#{$`}::#{type_name}" }
157 candidates << type_name
158
159 candidates.each do |candidate|
160 constant = ActiveSupport::Dependencies.safe_constantize(candidate)
161 if candidate == constant.to_s
162 @_type_candidates_cache[type_name] = candidate
163 return constant
164 end
165 end
166
167 raise NameError.new("uninitialized constant #{candidates.first}", candidates.first)
168 end
169 end

173 # Called by +instantiate+ to decide which class to use for a new
174 # record instance. For single-table inheritance, we check the record
175 # for a +type+ column and return the corresponding class.
176 def discriminate_class_for_record(record)
177 if using_single_table_inheritance?(record)
178 find_sti_class(record[inheritance_column])
179 else
180 super
181 end
182 end

184 def using_single_table_inheritance?(record)
185 record[inheritance_column].present? && has_attribute?(inheritance_column)
186 end
187
188 def find_sti_class(type_name)
189 type_name = base_class.type_for_attribute(inheritance_column).cast(type_name)
190 subclass = begin
191 if store_full_sti_class
192 ActiveSupport::Dependencies.constantize(type_name)
193 else
194 compute_type(type_name)
195 end
196 rescue NameError
197 raise SubclassNotFound,
198 "The single-table inheritance mechanism failed to locate the subclass: '#{type_name}'. " \
199 "This error is raised because the column '#{inheritance_column}' is reserved for storing the class in case of inheritance. " \
200 "Please rename this column if you didn't intend it to be used for storing the inheritance class " \
201 "or overwrite #{name}.inheritance_column to use another column for that information."
202 end
203 unless subclass == self || descendants.include?(subclass)
204 raise SubclassNotFound, "Invalid single-table inheritance type: #{subclass.name} is not a subclass of #{name}"
205 end
206 subclass
207 end

209 def type_condition(table = arel_table)
210 sti_column = arel_attribute(inheritance_column, table)
211 sti_names = ([self] + descendants).map(&:sti_name)
212
213 sti_column.in(sti_names)
214 end

216 # Detect the subclass from the inheritance column of attrs. If the inheritance column value
217 # is not self or a valid subclass, raises ActiveRecord::SubclassNotFound
218 def subclass_from_attributes(attrs)
219 attrs = attrs.to_h if attrs.respond_to?(:permitted?)
220 if attrs.is_a?(Hash)
221 subclass_name = attrs[inheritance_column] || attrs[inheritance_column.to_sym]
222
223 if subclass_name.present?
224 find_sti_class(subclass_name)
225 end
226 end

242 # Sets the attribute used for single table inheritance to this class name if this is not the
243 # ActiveRecord::Base descendant.
244 # Considering the hierarchy Reply < Message < ActiveRecord::Base, this makes it possible to
245 # do Reply.new without having to set <tt>Reply[Reply.inheritance_column] = "Reply"</tt> yourself.
246 # No such attribute would be set for objects of the Message class in that example.
247 def ensure_proper_type
248 klass = self.class
249 if klass.finder_needs_type_condition?
250 _write_attribute(klass.inheritance_column, klass.sti_name)
251 end
