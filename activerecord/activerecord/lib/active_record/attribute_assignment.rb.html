39 # Instantiates objects for all attribute classes that needs more than one constructor parameter. This is done
40 # by calling new on the column type or aggregation type (through composed_of) object with these parameters.
41 # So having the pairs written_on(1) = "2004", written_on(2) = "6", written_on(3) = "24", will instantiate
42 # written_on (a date type) with Date.new("2004", "6", "24"). You can also specify a typecast character in the
43 # parentheses to have the parameters typecasted before they're used in the constructor. Use i for Integer and
44 # f for Float. If all the values for a given attribute are empty, the attribute will be set to +nil+.
45 def assign_multiparameter_attributes(pairs)
46 execute_callstack_for_multiparameter_attributes(
47 extract_callstack_for_multiparameter_attributes(pairs)
48 )
49 end

51 def execute_callstack_for_multiparameter_attributes(callstack)
52 errors = []
53 callstack.each do |name, values_with_empty_parameters|
54 begin
55 if values_with_empty_parameters.each_value.all?(&:nil?)
56 values = nil
57 else
58 values = values_with_empty_parameters
59 end
60 send("#{name}=", values)
61 rescue => ex
62 errors << AttributeAssignmentError.new("error on assignment #{values_with_empty_parameters.values.inspect} to #{name} (#{ex.message})", ex, name)
63 end
64 end
65 unless errors.empty?
66 error_descriptions = errors.map(&:message).join(",")
67 raise MultiparameterAssignmentErrors.new(errors), "#{errors.size} error(s) on assignment of multiparameter attributes [#{error_descriptions}]"
68 end
69 end

71 def extract_callstack_for_multiparameter_attributes(pairs)
72 attributes = {}
73
74 pairs.each do |(multiparameter_name, value)|
75 attribute_name = multiparameter_name.split("(").first
76 attributes[attribute_name] ||= {}
77
78 parameter_value = value.empty? ? nil : type_cast_attribute_value(multiparameter_name, value)
79 attributes[attribute_name][find_parameter_position(multiparameter_name)] ||= parameter_value
80 end
81
82 attributes
83 end

87 end
