328 # Drops a table from the database.
329 #
330 # [<tt>:force</tt>]
331 # Set to +:cascade+ to drop dependent objects as well.
332 # Defaults to false.
333 # [<tt>:if_exists</tt>]
334 # Set to +true+ to only drop the table if it exists.
335 # Defaults to false.
336 # [<tt>:temporary</tt>]
337 # Set to +true+ to drop temporary table.
338 # Defaults to false.
339 #
340 # Although this command ignores most +options+ and the block if one is given,
341 # it can be helpful to provide these in a migration's +change+ method so it can be reverted.
342 # In that case, +options+ and the block will be used by create_table.
343 def drop_table(table_name, options = {})
344 execute "DROP#{' TEMPORARY' if options[:temporary]} TABLE#{' IF EXISTS' if options[:if_exists]} #{quote_table_name(table_name)}#{' CASCADE' if options[:force] == :cascade}"
345 end

394 def foreign_keys(table_name)
395 raise ArgumentError unless table_name.present?
396
397 scope = quoted_scope(table_name)
398
399 fk_info = exec_query(<<-SQL.strip_heredoc, "SCHEMA")
400 SELECT fk.referenced_table_name AS 'to_table',
401 fk.referenced_column_name AS 'primary_key',
402 fk.column_name AS 'column',
403 fk.constraint_name AS 'name',
404 rc.update_rule AS 'on_update',
405 rc.delete_rule AS 'on_delete'
406 FROM information_schema.key_column_usage fk
407 JOIN information_schema.referential_constraints rc
408 USING (constraint_schema, constraint_name)
409 WHERE fk.referenced_column_name IS NOT NULL
410 AND fk.table_schema = #{scope[:schema]}
411 AND fk.table_name = #{scope[:name]}
412 AND rc.table_name = #{scope[:name]}
413 SQL
414
415 fk_info.map do |row|
416 options = {
417 column: row["column"],
418 name: row["name"],
419 primary_key: row["primary_key"]
420 }
421
422 options[:on_update] = extract_foreign_key_action(row["on_update"])
423 options[:on_delete] = extract_foreign_key_action(row["on_delete"])
424
425 ForeignKeyDefinition.new(table_name, row["to_table"], options)
426 end
427 end

509 # In MySQL 5.7.5 and up, ONLY_FULL_GROUP_BY affects handling of queries that use
510 # DISTINCT and ORDER BY. It requires the ORDER BY columns in the select list for
511 # distinct queries, and requires that the ORDER BY include the distinct column.
512 # See https://dev.mysql.com/doc/refman/5.7/en/group-by-handling.html
513 def columns_for_distinct(columns, orders) # :nodoc:
514 order_columns = orders.reject(&:blank?).map { |s|
515 # Convert Arel node to string
516 s = s.to_sql unless s.is_a?(String)
517 # Remove any ASC/DESC modifiers
518 s.gsub(/\s+(?:ASC|DESC)\b/i, "")

522 end

709 # MySQL is too stupid to create a temporary table for use subquery, so we have
710 # to give it some prompting in the form of a subsubquery. Ugh!
711 def subquery_for(key, select)
712 subselect = select.clone
713 subselect.projections = [key]
714
715 # Materialize subquery by adding distinct
716 # to work with MySQL 5.7.6 which sets optimizer_switch='derived_merge=on'
717 subselect.distinct unless select.limit || select.offset || select.orders.any?
718
719 Arel::SelectManager.new(subselect.as("__active_record_temp")).project(Arel.sql(key.name))
720 end

726 def configure_connection
727 variables = @config.fetch(:variables, {}).stringify_keys
728
729 # By default, MySQL 'where id is null' selects the last inserted id; Turn this off.
730 variables["sql_auto_is_null"] = 0
731
732 # Increase timeout so the server doesn't disconnect us.
733 wait_timeout = self.class.type_cast_config_to_integer(@config[:wait_timeout])
734 wait_timeout = 2147483 unless wait_timeout.is_a?(Integer)
735 variables["wait_timeout"] = wait_timeout
736
737 defaults = [":default", :default].to_set
738
739 # Make MySQL reject illegal values rather than truncating or blanking them, see
740 # https://dev.mysql.com/doc/refman/5.7/en/sql-mode.html#sqlmode_strict_all_tables
741 # If the user has provided another value for sql_mode, don't replace it.
742 if sql_mode = variables.delete("sql_mode")
743 sql_mode = quote(sql_mode)
744 elsif !defaults.include?(strict_mode?)
745 if strict_mode?
746 sql_mode = "CONCAT(@@sql_mode, ',STRICT_ALL_TABLES')"
747 else
748 sql_mode = "REPLACE(@@sql_mode, 'STRICT_TRANS_TABLES', '')"
749 sql_mode = "REPLACE(#{sql_mode}, 'STRICT_ALL_TABLES', '')"
750 sql_mode = "REPLACE(#{sql_mode}, 'TRADITIONAL', '')"
751 end
752 sql_mode = "CONCAT(#{sql_mode}, ',NO_AUTO_VALUE_ON_ZERO')"
753 end
754 sql_mode_assignment = "@@SESSION.sql_mode = #{sql_mode}, " if sql_mode
755
756 # NAMES does not have an equals sign, see
757 # https://dev.mysql.com/doc/refman/5.7/en/set-names.html
758 # (trailing comma because variable_assignments will always have content)
759 if @config[:encoding]
760 encoding = "NAMES #{@config[:encoding]}".dup
761 encoding << " COLLATE #{@config[:collation]}" if @config[:collation]
762 encoding << ", "
763 end
764
765 # Gather up all of the SET variables...
766 variable_assignments = variables.map do |k, v|
767 if defaults.include?(v)
768 "@@SESSION.#{k} = DEFAULT" # Sets the value to the global or compile default
769 elsif !v.nil?
770 "@@SESSION.#{k} = #{quote(v)}"
771 end
772 # or else nil; compact to clear nils out
773 end.compact.join(", ")
774
775 # ...and send them all in one query
776 execute "SET #{encoding} #{sql_mode_assignment} #{variable_assignments}"
777 end

805 def integer_to_sql(limit) # :nodoc:
806 case limit
807 when 1; "tinyint"
808 when 2; "smallint"
809 when 3; "mediumint"
810 when nil, 4; "int"
811 when 5..8; "bigint"
812 else raise(ActiveRecordError, "No integer type has byte size #{limit}. Use a decimal with scale 0 instead.")
813 end
814 end

818 when 0..0xff; "tinytext"

820 when 0x10000..0xffffff; "mediumtext"
821 when 0x1000000..0xffffffff; "longtext"
822 else raise(ActiveRecordError, "No text type has byte length #{limit}")
823 end
824 end
