12 def mysql2_connection(config)
13 config = config.symbolize_keys
14 config[:flags] ||= 0
15
16 if config[:flags].kind_of? Array
17 config[:flags].push "FOUND_ROWS".freeze
18 else
19 config[:flags] |= Mysql2::Client::FOUND_ROWS
20 end

22 client = Mysql2::Client.new(config)
23 ConnectionAdapters::Mysql2Adapter.new(client, logger, nil, config)
