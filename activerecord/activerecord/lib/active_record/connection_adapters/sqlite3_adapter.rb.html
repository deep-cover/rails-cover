17 def sqlite3_connection(config)
18 # Require database.
19 unless config[:database]
20 raise ArgumentError, "No database file specified. Missing argument: database"
21 end
22
23 # Allow database path relative to Rails.root, but only if the database
24 # path is not the special path that tells sqlite to build a database only
25 # in memory.
26 if ":memory:" != config[:database]
27 config[:database] = File.expand_path(config[:database], Rails.root) if defined?(Rails.root)
28 dirname = File.dirname(config[:database])
29 Dir.mkdir(dirname) unless File.directory?(dirname)
30 end
31
32 db = SQLite3::Database.new(
33 config[:database].to_s,
34 results_as_hash: true
35 )
36
37 db.busy_timeout(ConnectionAdapters::SQLite3Adapter.type_cast_config_to_integer(config[:timeout])) if config[:timeout]
38
39 ConnectionAdapters::SQLite3Adapter.new(db, logger, nil, config)
40 rescue Errno::ENOENT => error
41 if error.message.include?("No such file or directory")
42 raise ActiveRecord::NoDatabaseError
43 else
44 raise
45 end
46 end
47 end

80 # and 0 or 't' and 'f'. Leaving <tt>ActiveRecord::ConnectionAdapters::SQLite3Adapter.represent_boolean_as_integer</tt>
81 # set to false is deprecated. SQLite databases have used 't' and 'f' to

209 def exec_query(sql, name = nil, binds = [], prepare: false)
210 type_casted_binds = type_casted_binds(binds)
211
212 log(sql, name, binds, type_casted_binds) do
213 ActiveSupport::Dependencies.interlock.permit_concurrent_loads do
214 # Don't cache statements if they are not prepared
215 unless prepare
216 stmt = @connection.prepare(sql)
217 begin
218 cols = stmt.columns
219 unless without_prepared_statement?(binds)
220 stmt.bind_params(type_casted_binds)
221 end
222 records = stmt.to_a
223 ensure
224 stmt.close
225 end
226 else
227 cache = @statements[sql] ||= {
228 stmt: @connection.prepare(sql)
229 }
230 stmt = cache[:stmt]
231 cols = cache[:cols] ||= stmt.columns
232 stmt.reset!
233 stmt.bind_params(type_casted_binds)
234 records = stmt.to_a
235 end
236
237 ActiveRecord::Result.new(cols, records)
238 end
239 end
240 end

405 def copy_table(from, to, options = {})
406 from_primary_key = primary_key(from)
407 from_primary_key_column = columns(from).select { |column| column.name == from_primary_key }.first
408 options[:id] = false
409 create_table(to, options) do |definition|
410 @definition = definition
411 @definition.primary_key(from_primary_key, from_primary_key_column.type) if from_primary_key.present?
412 columns(from).each do |column|
413 column_name = options[:rename] ?
414 (options[:rename][column.name] ||
415 options[:rename][column.name.to_sym] ||
416 column.name) : column.name
417 next if column_name == from_primary_key
418
419 @definition.column(column_name, column.type,
420 limit: column.limit, default: column.default,
421 precision: column.precision, scale: column.scale,
422 null: column.null, collation: column.collation)
423 end
424 yield @definition if block_given?
425 end
426 copy_table_indexes(from, to, options[:rename] || {})
427 copy_table_contents(from, to,
428 @definition.columns.map(&:name),
429 options[:rename] || {})
430 end

432 def copy_table_indexes(from, to, rename = {})
433 indexes(from).each do |index|
434 name = index.name
435 # indexes sqlite creates for internal use start with `sqlite_` and
436 # don't need to be copied
437 next if name.starts_with?("sqlite_")
438 if to == "a#{from}"
439 name = "t#{name}"
440 elsif from == "a#{to}"
441 name = name[1..-1]
442 end
443
444 to_column_names = columns(to).map(&:name)
445 columns = index.columns.map { |c| rename[c] || c }.select do |column|
446 to_column_names.include?(column)
447 end
448
449 unless columns.empty?
450 # index name can't be the same
451 opts = { name: name.gsub(/(^|_)(#{from})_/, "\\1#{to}_"), internal: true }
452 opts[:unique] = true if index.unique
453 add_index(to, columns, opts)
454 end
455 end
456 end

458 def copy_table_contents(from, to, columns, rename = {})
459 column_mappings = Hash[columns.map { |name| [name, name] }]
460 rename.each { |a| column_mappings[a.last] = a.first }
461 from_columns = columns(from).collect(&:name)
462 columns = columns.find_all { |col| from_columns.include?(column_mappings[col]) }
463 from_columns_to_copy = columns.map { |col| column_mappings[col] }
464 quoted_columns = columns.map { |col| quote_column_name(col) } * ","
465 quoted_from_columns = from_columns_to_copy.map { |col| quote_column_name(col) } * ","
466
467 exec_query("INSERT INTO #{quote_table_name(to)} (#{quoted_columns})
468 SELECT #{quoted_from_columns} FROM #{quote_table_name(from)}")
469 end

475 def translate_exception(exception, message)
476 case exception.message
477 # SQLite 3.8.2 returns a newly formatted error message:
478 # UNIQUE constraint failed: *table_name*.*column_name*
479 # Older versions of SQLite return:
480 # column *column_name* is not unique
481 when /column(s)? .* (is|are) not unique/, /UNIQUE constraint failed: .*/
482 RecordNotUnique.new(message)
483 when /.* may not be NULL/, /NOT NULL constraint failed: .*/
484 NotNullViolation.new(message)
485 when /FOREIGN KEY constraint failed/i
486 InvalidForeignKey.new(message)
487 else
488 super
489 end
490 end

494 def table_structure_with_collation(table_name, basic_structure)
495 collation_hash = {}
496 sql = <<-SQL
497 SELECT sql FROM
498 (SELECT * FROM sqlite_master UNION ALL
499 SELECT * FROM sqlite_temp_master)
500 WHERE type = 'table' AND name = #{quote(table_name)}
501 SQL
502
503 # Result will have following sample string
504 # CREATE TABLE "users" ("id" INTEGER PRIMARY KEY AUTOINCREMENT NOT NULL,
505 # "password_digest" varchar COLLATE "NOCASE");
506 result = exec_query(sql, "SCHEMA").first
507
508 if result
509 # Splitting with left parentheses and picking up last will return all
510 # columns separated with comma(,).
511 columns_string = result["sql"].split("(").last
512
513 columns_string.split(",").each do |column_string|
514 # This regex will match the column name and collation type and will save
515 # the value in $1 and $2 respectively.
516 collation_hash[$1] = $2 if COLLATE_REGEX =~ column_string
517 end
518
519 basic_structure.map! do |column|
520 column_name = column["name"]
521
522 if collation_hash.has_key? column_name
523 column["collation"] = collation_hash[column_name]
524 end
525
526 column
527 end
528 else
529 basic_structure.to_hash
530 end
531 end
