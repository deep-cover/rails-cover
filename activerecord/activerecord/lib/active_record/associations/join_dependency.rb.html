52 def self.walk_tree(associations, hash)
53 case associations
54 when Symbol, String
55 hash[associations.to_sym] ||= {}
56 when Array
57 associations.each do |assoc|
58 walk_tree assoc, hash
59 end
60 when Hash
61 associations.each do |k, v|
62 cache = hash[k] ||= {}
63 walk_tree v, cache
64 end
65 else
66 raise ConfigurationError, associations.inspect
67 end
68 end

70 # base is the base class on which operation is taking place.
71 # associations is the list of associations which are joined using hash, symbol or array.
72 # joins is the list of all string join commands and arel nodes.
73 #
74 # Example :
75 #
76 # class Physician < ActiveRecord::Base
77 # has_many :appointments
78 # has_many :patients, through: :appointments
79 # end
80 #
81 # If I execute `@physician.patients.to_a` then
82 # base # => Physician
83 # associations # => []
84 # joins # => [#<Arel::Nodes::InnerJoin: ...]
85 #
86 # However if I execute `Physician.joins(:appointments).to_a` then
87 # base # => Physician
88 # associations # => [:appointments]
89 # joins # => []
90 #

103 def join_constraints(joins_to_add, join_type)
104 joins = join_root.children.flat_map { |child|
105 make_join_constraints(join_root, child, join_type)

113 make_join_constraints(oj.join_root, child, join_type)

176 def make_join_constraints(parent, child, join_type, aliasing = false)
177 tables = aliasing ? table_aliases_for(parent, child) : child.tables
178 joins = make_constraints(parent, child, tables, join_type)
179
180 joins.concat child.children.flat_map { |c| make_join_constraints(child, c, join_type, aliasing) }
181 end

203 def walk(left, right)
204 intersection, missing = right.children.map { |node1|
205 [left.children.find { |node2| node1.match? node2 }, node1]
206 }.partition(&:first)
207
208 ojs = missing.flat_map { |_, n| make_outer_joins left, n }
209 intersection.flat_map { |l, r| walk l, r }.concat ojs
210 end

218 associations.map do |name, right|

232 def construct(ar_parent, parent, row, rs, seen, model_cache, aliases)
233 return if ar_parent.nil?
234
235 parent.children.each do |node|
236 if node.reflection.collection?
237 other = ar_parent.association(node.reflection.name)
238 other.loaded!
239 elsif ar_parent.association_cached?(node.reflection.name)
240 model = ar_parent.association(node.reflection.name).target
241 construct(model, node, row, rs, seen, model_cache, aliases)
242 next
243 end
244
245 key = aliases.column_alias(node, node.primary_key)
246 id = row[key]
247 if id.nil?
248 nil_association = ar_parent.association(node.reflection.name)
249 nil_association.loaded!
250 next
251 end
252
253 model = seen[ar_parent.object_id][node.base_klass][id]
254
255 if model
256 construct(model, node, row, rs, seen, model_cache, aliases)
257 else
258 model = construct_model(ar_parent, node, row, model_cache, id, aliases)
259
260 if node.reflection.scope &&
261 node.reflection.scope_for(node.base_klass.unscoped).readonly_value
262 model.readonly!
263 end
264
265 seen[ar_parent.object_id][node.base_klass][id] = model
266 construct(model, node, row, rs, seen, model_cache, aliases)
267 end
268 end
269 end

272 other = record.association(node.reflection.name)
