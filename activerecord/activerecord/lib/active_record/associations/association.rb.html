63 # The target is stale if the target no longer points to the record(s) that the
64 # relevant foreign_key(s) refers to. If stale, the association accessor method
65 # on the owner will reload the target. It's up to subclasses to implement the
66 # stale_state method if relevant.
67 #
68 # Note that if the target has not been loaded, it is not considered stale.
69 def stale_target?
70 !inversed && loaded? && @stale_state != stale_state
71 end

140 # Loads the \target if needed and returns it.
141 #
142 # This method is abstract in the sense that it relies on +find_target+,
143 # which is expected to be provided by descendants.
144 #
145 # If the \target is already \loaded it is just returned. Thus, you can call
146 # +load_target+ unconditionally to get the \target.
147 #
148 # ActiveRecord::RecordNotFound is rescued within the method, and it is
149 # not reraised. The proxy is \reset and +nil+ is the return value.
150 def load_target
151 @target = find_target if (@stale_state && stale_target?) || find_target?
152
153 loaded! unless loaded?
154 target
155 rescue ActiveRecord::RecordNotFound
156 reset
157 end

171 def initialize_attributes(record, except_from_scope_attributes = nil) #:nodoc:
172 except_from_scope_attributes ||= {}
173 skip_assign = [reflection.foreign_key, reflection.type].compact
174 assigned_keys = record.changed_attribute_names_to_save
175 assigned_keys += except_from_scope_attributes.keys.map(&:to_s)
176 attributes = scope_for_create.except!(*(assigned_keys - skip_assign))
177 record.send(:_assign_attributes, attributes) if attributes.any?
178 set_inverse_instance(record)
179 end

229 # Raises ActiveRecord::AssociationTypeMismatch unless +record+ is of
230 # the kind of the class of the associated objects. Meant to be used as
231 # a sanity check when you are about to assign an associated record.
232 def raise_on_type_mismatch!(record)
233 unless record.is_a?(reflection.klass)
234 fresh_class = reflection.class_name.safe_constantize
235 unless fresh_class && record.is_a?(fresh_class)
236 message = "#{reflection.class_name}(##{reflection.klass.object_id}) expected, "\
237 "got #{record.inspect} which is an instance of #{record.class}(##{record.class.object_id})"
238 raise ActiveRecord::AssociationTypeMismatch, message
239 end
240 end
