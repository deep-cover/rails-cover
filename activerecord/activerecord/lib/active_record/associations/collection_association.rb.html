5 # = Active Record Association Collection
6 #
7 # CollectionAssociation is an abstract class that provides common stuff to
8 # ease the implementation of association proxies that represent
9 # collections. See the class hierarchy in Association.
10 #
11 # CollectionAssociation:
12 # HasManyAssociation => has_many
13 # HasManyThroughAssociation + ThroughAssociation => has_many :through
14 #
15 # The CollectionAssociation class provides common methods to the collections
16 # defined by +has_and_belongs_to_many+, +has_many+ or +has_many+ with
17 # the +:through association+ option.
18 #
19 # You need to be careful with assumptions regarding the target: The proxy
20 # does not fetch records from the database until it needs them, but new
21 # ones created with +build+ are added to the target. So, the target may be
22 # non-empty and still lack children waiting to be read from the database.
23 # If you look directly to the database you cannot assume that's the entire
24 # collection because new records may have been added to the target, etc.
25 #
26 # If you need to work on all current children, new and existing records,
27 # +load_target+ and the +loaded+ flag are your friends.
28 class CollectionAssociation < Association #:nodoc:

202 # Returns the size of the collection by executing a SELECT COUNT(*)
203 # query if the collection hasn't been loaded, and calling
204 # <tt>collection.size</tt> if it has.
205 #
206 # If the collection has been already loaded +size+ and +length+ are
207 # equivalent. If not and you are going to need the records anyway
208 # +length+ will take one less query. Otherwise +size+ is more efficient.
209 #
210 # This method is abstract in the sense that it relies on
211 # +count_records+, which is a method descendants have to provide.
212 def size
213 if !find_target? || loaded?
214 target.size
215 elsif !association_scope.group_values.empty?
216 load_target.size
217 elsif !association_scope.distinct_value && target.is_a?(Array)
218 unsaved_records = target.select(&:new_record?)
219 unsaved_records.size + count_records
220 else
221 count_records
222 end
223 end

321 # We have some records loaded from the database (persisted) and some that are
322 # in-memory (memory). The same record may be represented in the persisted array
323 # and in the memory array.
324 #
325 # So the task of this method is to merge them according to the following rules:
326 #
327 # * The final array must not have duplicates
328 # * The order of the persisted array is to be preserved
329 # * Any changes made to attributes on objects in the memory array are to be preserved
330 # * Otherwise, attributes should have the value found in the database
331 def merge_target_lists(persisted, memory)
332 return persisted if memory.empty?
333 return memory if persisted.empty?
334
335 persisted.map! do |record|
336 if mem_record = memory.delete(record)
337
338 ((record.attribute_names & mem_record.attribute_names) - mem_record.changed_attribute_names_to_save).each do |name|
339 mem_record[name] = record[name]
340 end
341
342 mem_record
343 else
344 record
345 end
346 end
347
348 persisted + memory
349 end

448 def replace_on_target(record, index, skip_callbacks)
449 callback(:before_add, record) unless skip_callbacks
450
451 set_inverse_instance(record)
