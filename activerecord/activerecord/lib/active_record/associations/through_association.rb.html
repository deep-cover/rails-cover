11 # We merge in these scopes for two reasons:
12 #
13 # 1. To get the default_scope conditions for any of the other reflections in the chain
14 # 2. To get the type conditions for any STI models in the chain
15 def target_scope
16 scope = super
17 reflection.chain.drop(1).each do |reflection|
18 relation = reflection.klass.scope_for_association
19 scope.merge!(
20 relation.except(:select, :create_with, :includes, :preload, :joins, :eager_load)
21 )
22 end

26 # Construct attributes for :through pointing to owner and associate. This is used by the
27 # methods which create and delete records on the association.
28 #
29 # We only support indirectly modifying through associations which have a belongs_to source.
30 # This is the "has_many :tags, through: :taggings" situation, where the join model
31 # typically has a belongs_to on both side. In other words, associations which could also
32 # be represented as has_and_belongs_to_many associations.
33 #
34 # We do not support creating/deleting records on the association where the source has
35 # some other type, because this opens up a whole can of worms, and in basically any
36 # situation it is more natural for the user to just create or modify their join records
37 # directly as required.
38 def construct_join_attributes(*records)
39 ensure_mutable
40
41 if source_reflection.association_primary_key(reflection.klass) == reflection.klass.primary_key
42 join_attributes = { source_reflection.name => records }
43 else
44 join_attributes = {
45 source_reflection.foreign_key =>
46 records.map { |record|
47 record.send(source_reflection.association_primary_key(reflection.klass))
48 }
49 }
50 end
51
52 if options[:source_type]
53 join_attributes[source_reflection.foreign_type] =
54 records.map { |record| record.class.base_class.name }
55 end
56
57 if records.count == 1
58 Hash[join_attributes.map { |k, v| [k, v.first] }]
59 else
60 join_attributes
61 end
62 end

64 # Note: this does not capture all cases, for example it would be crazy to try to
65 # properly support stale-checking for nested associations.
66 def stale_state
67 if through_reflection.belongs_to?
68 owner[through_reflection.foreign_key] && owner[through_reflection.foreign_key].to_s
69 end

72 def foreign_key_present?

74 end

78 if reflection.has_one?
