150 def find(*ids) # :nodoc:
151 # We don't have cache keys for this stuff yet
152 return super unless ids.length == 1
153 return super if block_given? ||
154 primary_key.nil? ||
155 scope_attributes? ||
156 columns_hash.include?(inheritance_column)
157
158 id = ids.first
159
160 return super if StatementCache.unsupported_value?(id)
161
162 key = primary_key
163
164 statement = cached_find_by_statement(key) { |params|
165 where(key => params.bind).limit(1)
166 }
167
168 record = statement.execute([id], connection).first
169 unless record
170 raise RecordNotFound.new("Couldn't find #{name} with '#{primary_key}'=#{id}",
171 name, primary_key, id)
172 end
173 record
174 rescue ::RangeError
175 raise RecordNotFound.new("Couldn't find #{name} with an out of range value for '#{primary_key}'",
176 name, primary_key)
177 end

179 def find_by(*args) # :nodoc:
180 return super if scope_attributes? || reflect_on_all_aggregations.any?
181
182 hash = args.first
183
184 return super if !(Hash === hash) || hash.values.any? { |v|
185 StatementCache.unsupported_value?(v)
186 }
187
188 # We can't cache Post.find_by(author: david) ...yet
189 return super unless hash.keys.all? { |k| columns_hash.has_key?(k.to_s) }
190
191 keys = hash.keys
192
193 statement = cached_find_by_statement(keys) { |params|
194 wheres = keys.each_with_object({}) { |param, o|
195 o[param] = params.bind
196 }
197 where(wheres).limit(1)
198 }

200 statement.execute(hash.values, connection).first
201 rescue TypeError
202 raise ActiveRecord::StatementInvalid
203 rescue ::RangeError
204 nil
205 end

292 # New objects can be instantiated as either empty (pass no construction parameter) or pre-set with
293 # attributes but not yet saved (pass a hash with key names matching the associated table column names).
294 # In both instances, valid attribute keys are determined by the column names of the associated table --
295 # hence you can't have attributes that aren't part of the table columns.
296 #
297 # ==== Example:
298 # # Instantiates a single new object
299 # User.new(first_name: 'Jamie')
300 def initialize(attributes = nil)
301 self.class.define_attribute_methods
302 @attributes = self.class._default_attributes.deep_dup
303
304 init_internals
305 initialize_internals_callback
306
307 assign_attributes(attributes) if attributes
308
309 yield self if block_given?
310 _run_initialize_callbacks
311 end

346 # :method: clone
347 # Identical to Ruby's clone method. This is a "shallow" copy. Be warned that your attributes are not copied.
348 # That means that modifying attributes of the clone will modify the original, since they will both point to the
349 # same attributes hash. If you need a copy of your attributes hash, please use the #dup method.
350 #
351 # user = User.first
352 # new_user = user.clone
353 # user.name # => "Bob"
354 # new_user.name = "Joe"
355 # user.name # => "Joe"
356 #
357 # user.object_id == new_user.object_id # => false
358 # user.name.object_id == new_user.name.object_id # => true
359 #
360 # user.name.object_id == user.dup.name.object_id # => false
361
362 ##
363 # :method: dup
364 # Duped objects have no id assigned and are treated as new records. Note
365 # that this is a "shallow" copy as it copies the object's attributes
366 # only, not its associations. The extent of a "deep" copy is application
367 # specific and is therefore left to the application to implement according
368 # to its need.
369 # The dup method does not preserve the timestamps (created|updated)_(at|on).
370
371 ##

402 # Returns true if +comparison_object+ is the same exact object, or +comparison_object+
403 # is of the same type and +self+ has an ID and it is equal to +comparison_object.id+.
404 #
405 # Note that new records are different from any other record by definition, unless the
406 # other record is the receiver itself. Besides, if you fetch existing records with
407 # +select+ and leave the ID out, you're on your own, this predicate will return false.
408 #
409 # Note also that destroying a record preserves its ID in the model instance, so deleted
410 # models are still comparable.
411 def ==(comparison_object)
412 super ||
413 comparison_object.instance_of?(self.class) &&
414 !id.nil? &&
415 comparison_object.id == id
416 end
417 alias :eql? :==

483 # Takes a PP and prettily prints this record to it, allowing you to get a nice result from <tt>pp record</tt>
484 # when pp is required.
485 def pretty_print(pp)
486 return super if custom_inspect_method_defined?
487 pp.object_address_group(self) do
488 if defined?(@attributes) && @attributes
489 column_names = self.class.column_names.select { |name| has_attribute?(name) || new_record? }
490 pp.seplist(column_names, proc { pp.text "," }) do |column_name|
491 column_value = read_attribute(column_name)
492 pp.breakable " "
493 pp.group(1) do
494 pp.text column_name
495 pp.text ":"
496 pp.breakable
497 pp.pp column_value
498 end
499 end
500 else
501 pp.breakable " "
502 pp.text "not initialized"
503 end
504 end
