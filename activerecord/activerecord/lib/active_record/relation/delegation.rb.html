36 # This module creates compiled delegation methods dynamically at runtime, which makes
37 # subsequent calls to that method faster by avoiding method_missing. The delegations
38 # may vary depending on the klass of a relation, so we create a subclass of Relation

60 def delegate_to_scoped_klass(method)
61 @delegation_mutex.synchronize do
62 return if method_defined?(method)
63
64 if /\A[a-zA-Z_]\w*[!?]?\z/.match?(method)
65 module_eval <<-RUBY, __FILE__, __LINE__ + 1
66 def #{method}(*args, &block)
67 scoping { @klass.#{method}(*args, &block) }
68 end
69 RUBY
70 else
71 define_method method do |*args, &block|
72 scoping { @klass.public_send(method, *args, &block) }
73 end
74 end
75 end
76 end
77 end
