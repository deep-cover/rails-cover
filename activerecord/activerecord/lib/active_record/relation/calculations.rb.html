5 # Count the records.
6 #
7 # Person.count
8 # # => the total count of all people
9 #
10 # Person.count(:age)
11 # # => returns the total count of all people whose age is present in database
12 #
13 # Person.count(:all)
14 # # => performs a COUNT(*) (:all is an alias for '*')
15 #
16 # Person.distinct.count(:age)
17 # # => counts the number of different age values
18 #
19 # If #count is used with {Relation#group}[rdoc-ref:QueryMethods#group],
20 # it returns a Hash whose keys represent the aggregated column,
21 # and the values are the respective amounts:
22 #
23 # Person.group(:city).count
24 # # => { 'Rome' => 5, 'Paris' => 3 }
25 #
26 # If #count is used with {Relation#group}[rdoc-ref:QueryMethods#group] for multiple columns, it returns a Hash whose
27 # keys are an array containing the individual values of each column and the value
28 # of each key would be the #count.
29 #
30 # Article.group(:status, :category).count
31 # # => {["draft", "business"]=>10, ["draft", "technology"]=>4,
32 # ["published", "business"]=>0, ["published", "technology"]=>2}
33 #
34 # If #count is used with {Relation#select}[rdoc-ref:QueryMethods#select], it will count the selected columns:
35 #
36 # Person.select(:age).count
37 # # => counts the number of different age values
38 #
39 # Note: not all valid {Relation#select}[rdoc-ref:QueryMethods#select] expressions are valid #count expressions. The specifics differ
40 # between databases. In invalid cases, an error from the database is thrown.

100 # This calculates aggregate values in the given column. Methods for #count, #sum, #average,
101 # #minimum, and #maximum have been added as shortcuts.
102 #
103 # Person.calculate(:count, :all) # The same as Person.count
104 # Person.average(:age) # SELECT AVG(age) FROM people...
105 #
106 # # Selects the minimum age for any family without any minors
107 # Person.group(:last_name).having("min(age) > 17").minimum(:age)
108 #
109 # Person.sum("2 * age")
110 #
111 # There are two basic forms of output:
112 #
113 # * Single aggregate value: The single value is type cast to Integer for COUNT, Float
114 # for AVG, and the given column's type for everything else.
115 #
116 # * Grouped values: This returns an ordered hash of the values and groups them. It
117 # takes either a column name, or the name of a belongs_to association.
118 #
119 # values = Person.group('last_name').maximum(:age)
120 # puts values["Drake"]
121 # # => 43
122 #
123 # drake = Family.find_by(last_name: 'Drake')
124 # values = Person.group(:family).maximum(:age) # Person belongs_to :family
125 # puts values[drake]
126 # # => 43
127 #
128 # values.each do |family, max_age|
129 # ...
130 # end
131 def calculate(operation, column_name)
132 if has_include?(column_name)
133 relation = apply_join_dependency
134 relation.distinct! if operation.to_s.downcase == "count"
135
136 relation.calculate(operation, column_name)
137 else
138 perform_calculation(operation, column_name)
139 end
140 end

142 # Use #pluck as a shortcut to select one or more attributes without
143 # loading a bunch of records just to grab the attributes you want.
144 #
145 # Person.pluck(:name)
146 #
147 # instead of
148 #
149 # Person.all.map(&:name)
150 #
151 # Pluck returns an Array of attribute values type-casted to match
152 # the plucked column names, if they can be deduced. Plucking an SQL fragment
153 # returns String values by default.
154 #
155 # Person.pluck(:name)
156 # # SELECT people.name FROM people
157 # # => ['David', 'Jeremy', 'Jose']
158 #
159 # Person.pluck(:id, :name)
160 # # SELECT people.id, people.name FROM people
161 # # => [[1, 'David'], [2, 'Jeremy'], [3, 'Jose']]
162 #
163 # Person.distinct.pluck(:role)
164 # # SELECT DISTINCT role FROM people
165 # # => ['admin', 'member', 'guest']
166 #
167 # Person.where(age: 21).limit(5).pluck(:id)
168 # # SELECT people.id FROM people WHERE people.age = 21 LIMIT 5
169 # # => [2, 3]
170 #
171 # Person.pluck('DATEDIFF(updated_at, created_at)')
172 # # SELECT DATEDIFF(updated_at, created_at) FROM people
173 # # => ['0', '27761', '173']
174 #
175 # See also #ids.
176 #
177 def pluck(*column_names)
178 if loaded? && (column_names.map(&:to_s) - @klass.attribute_names - @klass.attribute_aliases.keys).empty?
179 return records.pluck(*column_names)
180 end
181
182 if has_include?(column_names.first)
183 relation = apply_join_dependency
184 relation.pluck(*column_names)
185 else
186 enforce_raw_sql_whitelist(column_names)
187 relation = spawn
188 relation.select_values = column_names.map { |cn|
189 @klass.has_attribute?(cn) || @klass.attribute_alias?(cn) ? arel_attribute(cn) : cn
190 }
191 result = skip_query_cache_if_necessary { klass.connection.select_all(relation.arel, nil) }
192 result.cast_values(klass.attribute_types)
193 end
194 end

210 def perform_calculation(operation, column_name)
211 operation = operation.to_s.downcase
212
213 # If #count is used with #distinct (i.e. `relation.distinct.count`) it is
214 # considered distinct.
215 distinct = distinct_value
216
217 if operation == "count"
218 column_name ||= select_for_count
219 if column_name == :all
220 if distinct && (group_values.any? || !(has_limit_or_offset? && order_values.any?))
221 column_name = primary_key
222 end
223 elsif column_name =~ /\s*DISTINCT[\s(]+/i
224 distinct = nil
225 end
226 end
227
228 if group_values.any?
229 execute_grouped_calculation(operation, column_name, distinct)
230 else
231 execute_simple_calculation(operation, column_name, distinct)
232 end
233 end

235 def aggregate_column(column_name)
236 return column_name if Arel::Expressions === column_name
237
238 if @klass.has_attribute?(column_name.to_s) || @klass.attribute_alias?(column_name.to_s)
239 @klass.arel_attribute(column_name)
240 else
241 Arel.sql(column_name == :all ? "*" : column_name.to_s)
242 end
243 end

245 def operation_over_aggregate_column(column, operation, distinct)
246 operation == "count" ? column.count(distinct) : column.send(operation)
247 end

249 def execute_simple_calculation(operation, column_name, distinct) #:nodoc:
250 column_alias = column_name
251
252 if operation == "count" && has_limit_or_offset?
253 # Shortcut when limit is zero.
254 return 0 if limit_value == 0
255
256 query_builder = build_count_subquery(spawn, column_name, distinct)
257 else
258 # PostgreSQL doesn't like ORDER BY when there are no GROUP BY
259 relation = unscope(:order).distinct!(false)
260
261 column = aggregate_column(column_name)
262
263 select_value = operation_over_aggregate_column(column, operation, distinct)
264 if operation == "sum" && distinct
265 select_value.distinct = true
266 end
267
268 column_alias = select_value.alias
269 column_alias ||= @klass.connection.column_name_for_operation(operation, select_value)
270 relation.select_values = [select_value]
271
272 query_builder = relation.arel
273 end
274
275 result = skip_query_cache_if_necessary { @klass.connection.select_all(query_builder, nil) }
276 row = result.first
277 value = row && row.values.first
278 type = result.column_types.fetch(column_alias) do
279 type_for(column_name)
280 end
281
282 type_cast_calculated_value(value, type, operation)
283 end

285 def execute_grouped_calculation(operation, column_name, distinct) #:nodoc:
286 group_attrs = group_values
287
288 if group_attrs.first.respond_to?(:to_sym)
289 association = @klass._reflect_on_association(group_attrs.first)
290 associated = group_attrs.size == 1 && association && association.belongs_to? # only count belongs_to associations
291 group_fields = Array(associated ? association.foreign_key : group_attrs)
292 else
293 group_fields = group_attrs
294 end
295 group_fields = arel_columns(group_fields)
296
297 group_aliases = group_fields.map { |field| column_alias_for(field) }
298 group_columns = group_aliases.zip(group_fields)
299
300 if operation == "count" && column_name == :all
301 aggregate_alias = "count_all"
302 else
303 aggregate_alias = column_alias_for([operation, column_name].join(" "))
304 end
305
306 select_values = [
307 operation_over_aggregate_column(
308 aggregate_column(column_name),
309 operation,
310 distinct).as(aggregate_alias)
311 ]
312 select_values += self.select_values unless having_clause.empty?
313
314 select_values.concat group_columns.map { |aliaz, field|
315 if field.respond_to?(:as)
316 field.as(aliaz)
317 else
318 "#{field} AS #{aliaz}"
319 end
320 }
321
322 relation = except(:group).distinct!(false)
323 relation.group_values = group_fields
324 relation.select_values = select_values
325
326 calculated_data = skip_query_cache_if_necessary { @klass.connection.select_all(relation.arel, nil) }
327
328 if association
329 key_ids = calculated_data.collect { |row| row[group_aliases.first] }
330 key_records = association.klass.base_class.where(association.klass.base_class.primary_key => key_ids)
331 key_records = Hash[key_records.map { |r| [r.id, r] }]
332 end
333
334 Hash[calculated_data.map do |row|
335 key = group_columns.map { |aliaz, col_name|
336 type = type_for(col_name) do
337 calculated_data.column_types.fetch(aliaz, Type.default_value)
338 end
339 type_cast_calculated_value(row[aliaz], type)
340 }
341 key = key.first if key.size == 1
342 key = key_records[key] if associated
343
344 type = calculated_data.column_types.fetch(aggregate_alias) { type_for(column_name) }
345 [key, type_cast_calculated_value(row[aggregate_alias], type, operation)]
346 end]
347 end

349 # Converts the given keys to the value that the database adapter returns as
350 # a usable column name:
351 #
352 # column_alias_for("users.id") # => "users_id"
353 # column_alias_for("sum(id)") # => "sum_id"
354 # column_alias_for("count(distinct users.id)") # => "count_distinct_users_id"
355 # column_alias_for("count(*)") # => "count_all"
356 def column_alias_for(keys)
357 if keys.respond_to? :name
358 keys = "#{keys.relation.name}.#{keys.name}"
359 end
360
361 table_name = keys.to_s.downcase
362 table_name.gsub!(/\*/, "all")
363 table_name.gsub!(/\W+/, " ")
364 table_name.strip!
365 table_name.gsub!(/ +/, "_")
366
367 @klass.connection.table_alias_for(table_name)
368 end

370 def type_for(field, &block)
371 field_name = field.respond_to?(:name) ? field.name.to_s : field.to_s.split(".").last
372 @klass.type_for_attribute(field_name, &block)
373 end

375 def type_cast_calculated_value(value, type, operation = nil)
376 case operation
377 when "count" then value.to_i
378 when "sum" then type.deserialize(value || 0)
379 when "average" then value.respond_to?(:to_d) ? value.to_d : value
380 else type.deserialize(value)
381 end
382 end

384 def select_for_count
385 if select_values.present?
386 return select_values.first if select_values.one?
387 select_values.join(", ")
388 else
389 :all
390 end

393 def build_count_subquery(relation, column_name, distinct)
394 relation.select_values = [
395 if column_name == :all
396 distinct ? table[Arel.star] : Arel.sql(FinderMethods::ONE_AS_ONE)
397 else
398 column_alias = Arel.sql("count_column")
399 aggregate_column(column_name).as(column_alias)
400 end
401 ]
402
403 subquery = relation.arel.as(Arel.sql("subquery_for_count"))
404 select_value = operation_over_aggregate_column(column_alias || Arel.star, "count", false)
405
406 Arel::SelectManager.new(subquery).project(select_value)
407 end
