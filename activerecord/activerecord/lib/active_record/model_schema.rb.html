136 # Derives the join table name for +first_table+ and +second_table+. The
137 # table names appear in alphabetical order. A common prefix is removed
138 # (useful for namespaced models like Music::Artist and Music::Record):
139 #
140 # artists, records => artists_records
141 # records, artists => artists_records
142 # music_artists, music_records => music_artists_records
143 def self.derive_join_table_name(first_table, second_table) # :nodoc:
144 [first_table.to_s, second_table.to_s].sort.join("\0").gsub(/^(.*_)(.+)\0\1(.+)/, '\1\2_\3').tr("\0", "_")
145 end

148 # Guesses the table name (in forced lower-case) based on the name of the class in the
149 # inheritance hierarchy descending directly from ActiveRecord::Base. So if the hierarchy
150 # looks like: Reply < Message < ActiveRecord::Base, then Message is used
151 # to guess the table name even when called on Reply. The rules used to do the guess
152 # are handled by the Inflector class in Active Support, which knows almost all common
153 # English inflections. You can add new inflections in config/initializers/inflections.rb.
154 #
155 # Nested classes are given table names prefixed by the singular form of
156 # the parent's table name. Enclosing modules are not considered.
157 #
158 # ==== Examples
159 #
160 # class Invoice < ActiveRecord::Base
161 # end
162 #
163 # file class table_name
164 # invoice.rb Invoice invoices
165 #
166 # class Invoice < ActiveRecord::Base
167 # class Lineitem < ActiveRecord::Base
168 # end
169 # end
170 #
171 # file class table_name
172 # invoice.rb Invoice::Lineitem invoice_lineitems
173 #
174 # module Invoice
175 # class Lineitem < ActiveRecord::Base
176 # end
177 # end
178 #
179 # file class table_name
180 # invoice/lineitem.rb Invoice::Lineitem lineitems
181 #
182 # Additionally, the class-level +table_name_prefix+ is prepended and the
183 # +table_name_suffix+ is appended. So if you have "myapp_" as a prefix,
184 # the table name guess for an Invoice class becomes "myapp_invoices".
185 # Invoice::Lineitem becomes "myapp_invoice_lineitems".
186 #
187 # You can also set your own table name explicitly:
188 #
189 # class Mouse < ActiveRecord::Base
190 # self.table_name = "mice"
191 # end
192 def table_name
193 reset_table_name unless defined?(@table_name)
194 @table_name
195 end

498 # Computes and returns a table name according to default conventions.
499 def compute_table_name
500 base = base_class
501 if self == base
502 # Nested classes are prefixed with singular parent table name.
503 if parent < Base && !parent.abstract_class?
504 contained = parent.table_name
505 contained = contained.singularize if parent.pluralize_table_names
506 contained += "_"
507 end
508
509 "#{full_table_name_prefix}#{contained}#{undecorated_table_name(name)}#{full_table_name_suffix}"
510 else
511 # STI subclasses always use their superclass' table.
512 base.table_name
513 end
514 end
