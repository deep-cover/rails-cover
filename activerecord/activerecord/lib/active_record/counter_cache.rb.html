9 # Resets one or more counter caches to their correct value using an SQL
10 # count query. This is useful when adding new counter caches, or if the
11 # counter has been corrupted or modified directly by SQL.
12 #
13 # ==== Parameters
14 #
15 # * +id+ - The id of the object you wish to reset a counter on.
16 # * +counters+ - One or more association counters to reset. Association name or counter name can be given.
17 # * <tt>:touch</tt> - Touch timestamp columns when updating.
18 # Pass +true+ to touch +updated_at+ and/or +updated_on+. Pass a symbol to
19 # touch that column or an array of symbols to touch just those ones.
20 #
21 # ==== Examples
22 #
23 # # For the Post with id #1, reset the comments_count
24 # Post.reset_counters(1, :comments)
25 #
26 # # Like above, but also touch the +updated_at+ and/or +updated_on+
27 # # attributes.
28 # Post.reset_counters(1, :comments, touch: true)
29 def reset_counters(id, *counters, touch: nil)
30 object = find(id)
31
32 counters.each do |counter_association|
33 has_many_association = _reflect_on_association(counter_association)
34 unless has_many_association
35 has_many = reflect_on_all_associations(:has_many)
36 has_many_association = has_many.find { |association| association.counter_cache_column && association.counter_cache_column.to_sym == counter_association.to_sym }
37 counter_association = has_many_association.plural_name if has_many_association
38 end
39 raise ArgumentError, "'#{name}' has no association called '#{counter_association}'" unless has_many_association
40
41 if has_many_association.is_a? ActiveRecord::Reflection::ThroughReflection
42 has_many_association = has_many_association.through_reflection
43 end
44
45 foreign_key = has_many_association.foreign_key.to_s
46 child_class = has_many_association.klass
47 reflection = child_class._reflections.values.find { |e| e.belongs_to? && e.foreign_key.to_s == foreign_key && e.options[:counter_cache].present? }
48 counter_name = reflection.counter_cache_column
49
50 updates = { counter_name.to_sym => object.send(counter_association).count(:all) }
51 updates.merge!(touch_updates(touch)) if touch
52

54 end

59 # A generic "counter updater" implementation, intended primarily to be
60 # used by #increment_counter and #decrement_counter, but which may also
61 # be useful on its own. It simply does a direct SQL update for the record
62 # with the given ID, altering the given hash of counters by the amount
63 # given by the corresponding value:
64 #
65 # ==== Parameters
66 #
67 # * +id+ - The id of the object you wish to update a counter on or an array of ids.
68 # * +counters+ - A Hash containing the names of the fields
69 # to update as keys and the amount to update the field by as values.
70 # * <tt>:touch</tt> option - Touch timestamp columns when updating.
71 # Pass +true+ to touch +updated_at+ and/or +updated_on+. Pass a symbol to
72 # touch that column or an array of symbols to touch just those ones.
73 #
74 # ==== Examples
75 #
76 # # For the Post with id of 5, decrement the comment_count by 1, and
77 # # increment the action_count by 1
78 # Post.update_counters 5, comment_count: -1, action_count: 1
79 # # Executes the following SQL:
80 # # UPDATE posts
81 # # SET comment_count = COALESCE(comment_count, 0) - 1,
82 # # action_count = COALESCE(action_count, 0) + 1
83 # # WHERE id = 5
84 #
85 # # For the Posts with id of 10 and 15, increment the comment_count by 1
86 # Post.update_counters [10, 15], comment_count: 1
87 # # Executes the following SQL:
88 # # UPDATE posts
89 # # SET comment_count = COALESCE(comment_count, 0) + 1
90 # # WHERE id IN (10, 15)
91 #
92 # # For the Posts with id of 10 and 15, increment the comment_count by 1
93 # # and update the updated_at value for each counter.
94 # Post.update_counters [10, 15], comment_count: 1, touch: true
95 # # Executes the following SQL:
96 # # UPDATE posts
97 # # SET comment_count = COALESCE(comment_count, 0) + 1,
98 # # `updated_at` = '2016-10-13T09:59:23-05:00'
99 # # WHERE id IN (10, 15)
100 def update_counters(id, counters)
101 touch = counters.delete(:touch)
102
103 updates = counters.map do |counter_name, value|
104 operator = value < 0 ? "-" : "+"
105 quoted_column = connection.quote_column_name(counter_name)
106 "#{quoted_column} = COALESCE(#{quoted_column}, 0) #{operator} #{value.abs}"
107 end
108
109 if touch
110 touch_updates = touch_updates(touch)
111 updates << sanitize_sql_for_assignment(touch_updates) unless touch_updates.empty?
112 end
113
114 unscoped.where(primary_key => id).update_all updates.join(", ")
115 end

117 # Increment a numeric field by one, via a direct SQL update.
118 #
119 # This method is used primarily for maintaining counter_cache columns that are
120 # used to store aggregate values. For example, a +DiscussionBoard+ may cache
121 # posts_count and comments_count to avoid running an SQL query to calculate the
122 # number of posts and comments there are, each time it is displayed.
123 #
124 # ==== Parameters
125 #
126 # * +counter_name+ - The name of the field that should be incremented.
127 # * +id+ - The id of the object that should be incremented or an array of ids.
128 # * <tt>:touch</tt> - Touch timestamp columns when updating.
129 # Pass +true+ to touch +updated_at+ and/or +updated_on+. Pass a symbol to
130 # touch that column or an array of symbols to touch just those ones.
131 #
132 # ==== Examples
133 #
134 # # Increment the posts_count column for the record with an id of 5
135 # DiscussionBoard.increment_counter(:posts_count, 5)
136 #
137 # # Increment the posts_count column for the record with an id of 5
138 # # and update the updated_at value.
139 # DiscussionBoard.increment_counter(:posts_count, 5, touch: true)
140 def increment_counter(counter_name, id, touch: nil)
141 update_counters(id, counter_name => 1, touch: touch)
142 end

144 # Decrement a numeric field by one, via a direct SQL update.
145 #
146 # This works the same as #increment_counter but reduces the column value by
147 # 1 instead of increasing it.
148 #
149 # ==== Parameters
150 #
151 # * +counter_name+ - The name of the field that should be decremented.
152 # * +id+ - The id of the object that should be decremented or an array of ids.
153 # * <tt>:touch</tt> - Touch timestamp columns when updating.
154 # Pass +true+ to touch +updated_at+ and/or +updated_on+. Pass a symbol to
155 # touch that column or an array of symbols to touch just those ones.
156 #
157 # ==== Examples
158 #
159 # # Decrement the posts_count column for the record with an id of 5
160 # DiscussionBoard.decrement_counter(:posts_count, 5)
161 #
162 # # Decrement the posts_count column for the record with an id of 5
163 # # and update the updated_at value.
164 # DiscussionBoard.decrement_counter(:posts_count, 5, touch: true)
165 def decrement_counter(counter_name, id, touch: nil)
166 update_counters(id, counter_name => -1, touch: touch)
167 end
