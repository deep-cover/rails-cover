16 # Instrument the given block by measuring the time taken to execute it
17 # and publish it. Notice that events get sent even if an error occurs
18 # in the passed-in block.
19 def instrument(name, payload = {})
20 # some of the listeners might have state
21 listeners_state = start name, payload
22 begin
23 yield payload
24 rescue Exception => e
25 payload[:exception] = [e.class.name, e.message]
26 payload[:exception_object] = e
27 raise e
28 ensure
29 finish_with_state listeners_state, name, payload
30 end
31 end
32
33 # Send a start notification with +name+ and +payload+.
34 def start(name, payload)
35 @notifier.start name, @id, payload
36 end
37
38 # Send a finish notification with +name+ and +payload+.
39 def finish(name, payload)
40 @notifier.finish name, @id, payload
41 end
42
43 def finish_with_state(listeners_state, name, payload)
44 @notifier.finish name, @id, payload, listeners_state
45 end
