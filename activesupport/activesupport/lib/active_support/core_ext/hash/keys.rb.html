4 # Returns a new hash with all keys converted using the +block+ operation.
5 #
6 # hash = { name: 'Rob', age: '28' }
7 #
8 # hash.transform_keys { |key| key.to_s.upcase } # => {"NAME"=>"Rob", "AGE"=>"28"}

10 # If you do not provide a +block+, it will return an Enumerator
11 # for chaining with other methods:
12 #
13 # hash.transform_keys.with_index { |k, i| [k, i].join } # => {"name0"=>"Rob", "age1"=>"28"}
14 def transform_keys
15 return enum_for(:transform_keys) { size } unless block_given?
16 result = {}
17 each_key do |key|
18 result[yield(key)] = self[key]
19 end
20 result
21 end unless method_defined? :transform_keys

68 # Validates all keys in a hash match <tt>*valid_keys</tt>, raising
69 # +ArgumentError+ on a mismatch.
70 #
71 # Note that keys are treated differently than HashWithIndifferentAccess,
72 # meaning that string and symbol keys will not match.
73 #
74 # { name: 'Rob', years: '28' }.assert_valid_keys(:name, :age) # => raises "ArgumentError: Unknown key: :years. Valid keys are: :name, :age"
75 # { name: 'Rob', age: '28' }.assert_valid_keys('name', 'age') # => raises "ArgumentError: Unknown key: :name. Valid keys are: 'name', 'age'"
76 # { name: 'Rob', age: '28' }.assert_valid_keys(:name, :age) # => passes, raises nothing
77 def assert_valid_keys(*valid_keys)
78 valid_keys.flatten!
79 each_key do |k|
80 unless valid_keys.include?(k)
81 raise ArgumentError.new("Unknown key: #{k.inspect}. Valid keys are: #{valid_keys.map(&:inspect).join(', ')}")
82 end
83 end
84 end

145 def _deep_transform_keys_in_object(object, &block)
146 case object
147 when Hash
148 object.each_with_object({}) do |(key, value), result|
149 result[yield(key)] = _deep_transform_keys_in_object(value, &block)
150 end
151 when Array
152 object.map { |e| _deep_transform_keys_in_object(e, &block) }
153 else
154 object
155 end
156 end

158 def _deep_transform_keys_in_object!(object, &block)
159 case object
160 when Hash
161 object.keys.each do |key|
162 value = object.delete(key)
163 object[yield(key)] = _deep_transform_keys_in_object!(value, &block)
164 end
165 object
166 when Array
167 object.map! { |e| _deep_transform_keys_in_object!(e, &block) }
168 else
169 object
170 end
