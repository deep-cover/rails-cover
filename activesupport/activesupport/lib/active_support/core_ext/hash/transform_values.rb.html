4 # Returns a new hash with the results of running +block+ once for every value.
5 # The keys are unchanged.
6 #
7 # { a: 1, b: 2, c: 3 }.transform_values { |x| x * 2 } # => { a: 2, b: 4, c: 6 }
8 #
9 # If you do not provide a +block+, it will return an Enumerator
10 # for chaining with other methods:
11 #
12 # { a: 1, b: 2 }.transform_values.with_index { |v, i| [v, i].join.to_i } # => { a: 10, b: 21 }
13 def transform_values
14 return enum_for(:transform_values) { size } unless block_given?
15 return {} if empty?

17 each do |key, value|
18 result[key] = yield(value)
