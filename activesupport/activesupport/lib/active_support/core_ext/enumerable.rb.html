15 # Calculates a sum from the elements.
16 #
17 # payments.sum { |p| p.price * p.tax_rate }
18 # payments.sum(&:price)
19 #
20 # The latter is a shortcut for:
21 #
22 # payments.inject(0) { |sum, p| sum + p.price }
23 #
24 # It can also calculate the sum without the use of a block.
25 #
26 # [5, 15, 10].sum # => 30
27 # ['foo', 'bar'].sum # => "foobar"
28 # [[1, 2], [3, 1, 5]].sum # => [1, 2, 3, 1, 5]
29 #
30 # The default sum of an empty list is zero. You can override this default:
31 #
32 # [].sum(Payment.new(0)) { |i| i.amount } # => Payment.new(0)
33 def sum(identity = nil, &block)
34 if identity
35 _original_sum_with_required_identity(identity, &block)
36 elsif block_given?
37 map(&block).sum(identity)
38 else
39 inject(:+) || 0
40 end
41 end
42 else
43 def sum(identity = nil, &block)
44 if block_given?
45 map(&block).sum(identity)
46 else
47 sum = identity ? inject(identity, :+) : inject(:+)
48 sum || identity || 0
49 end
50 end
51 end

69 # Returns +true+ if the enumerable has more than 1 element. Functionally
70 # equivalent to <tt>enum.to_a.size > 1</tt>. Can be called with a block too,
71 # much like any?, so <tt>people.many? { |p| p.age > 26 }</tt> returns +true+
72 # if more than one person is over 26.
73 def many?
74 cnt = 0
75 if block_given?
76 any? do |element|
77 cnt += 1 if yield element
78 cnt > 1
79 end
80 else
81 any? { (cnt += 1) > 1 }
82 end
83 end

91 # Returns a copy of the enumerable without the specified elements.
92 #
93 # ["David", "Rafael", "Aaron", "Todd"].without "Aaron", "Todd"
94 # # => ["David", "Rafael"]
95 #
96 # {foo: 1, bar: 2, baz: 3}.without :bar
97 # # => {foo: 1, baz: 3}
98 def without(*elements)
99 reject { |element| elements.include?(element) }
100 end

102 # Convert an enumerable to an array based on the given key.
103 #
104 # [{ name: "David" }, { name: "Rafael" }, { name: "Aaron" }].pluck(:name)
105 # # => ["David", "Rafael", "Aaron"]
106 #
107 # [{ id: 1, name: "David" }, { id: 2, name: "Rafael" }].pluck(:id, :name)
108 # # => [[1, "David"], [2, "Rafael"]]
109 def pluck(*keys)
110 if keys.many?
111 map { |element| keys.map { |key| element[key] } }
112 else
113 map { |element| element[keys.first] }
114 end
115 end

119 # Optimize range sum to use arithmetic progression if a block is not given and
120 # we have a range of numeric values.
121 def sum(identity = nil)
122 if block_given? || !(first.is_a?(Integer) && last.is_a?(Integer))
123 super
124 else
125 actual_last = exclude_end? ? (last - 1) : last
126 if actual_last >= first
127 sum = identity || 0
