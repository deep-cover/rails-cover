6# Extends the module object with class/module and instance accessors for
7# class/module attributes, just like the native attr* accessors for instance
8# attributes.
9class Module
10 # Defines a class attribute and creates a class and instance reader methods.
11 # The underlying class variable is set to +nil+, if it is not previously
12 # defined. All class and instance methods created will be public, even if
13 # this method is called with a private or protected access modifier.
14 #
15 # module HairColors
16 # mattr_reader :hair_colors
17 # end
18 #
19 # HairColors.hair_colors # => nil
20 # HairColors.class_variable_set("@@hair_colors", [:brown, :black])
21 # HairColors.hair_colors # => [:brown, :black]
22 #
23 # The attribute name must be a valid method name in Ruby.
24 #
25 # module Foo
26 # mattr_reader :"1_Badname"
27 # end
28 # # => NameError: invalid attribute name: 1_Badname
29 #
30 # If you want to opt out the creation on the instance reader method, pass
31 # <tt>instance_reader: false</tt> or <tt>instance_accessor: false</tt>.
32 #
33 # module HairColors
34 # mattr_reader :hair_colors, instance_reader: false
35 # end
36 #
37 # class Person
38 # include HairColors
39 # end
40 #
41 # Person.new.hair_colors # => NoMethodError
42 #
43 # You can set a default value for the attribute.
44 #
45 # module HairColors
46 # mattr_reader :hair_colors, default: [:brown, :black, :blonde, :red]
47 # end
48 #
49 # class Person
50 # include HairColors
51 # end
52 #
53 # Person.new.hair_colors # => [:brown, :black, :blonde, :red]
54 def mattr_reader(*syms, instance_reader: true, instance_accessor: true, default: nil)
55 syms.each do |sym|
56 raise NameError.new("invalid attribute name: #{sym}") unless /\A[_A-Za-z]\w*\z/.match?(sym)
57 class_eval(<<-EOS, __FILE__, __LINE__ + 1)
58 @@#{sym} = nil unless defined? @@#{sym}
59
60 def self.#{sym}
61 @@#{sym}
62 end
63 EOS
64
65 if instance_reader && instance_accessor
66 class_eval(<<-EOS, __FILE__, __LINE__ + 1)
67 def #{sym}
68 @@#{sym}
69 end
70 EOS
71 end
72
73 sym_default_value = (block_given? && default.nil?) ? yield : default
74 class_variable_set("@@#{sym}", sym_default_value) unless sym_default_value.nil?
75 end
76 end
77 alias :cattr_reader :mattr_reader

79 # Defines a class attribute and creates a class and instance writer methods to
80 # allow assignment to the attribute. All class and instance methods created
81 # will be public, even if this method is called with a private or protected
82 # access modifier.
83 #
84 # module HairColors
85 # mattr_writer :hair_colors
86 # end
87 #
88 # class Person
89 # include HairColors
90 # end
91 #
92 # HairColors.hair_colors = [:brown, :black]
93 # Person.class_variable_get("@@hair_colors") # => [:brown, :black]
94 # Person.new.hair_colors = [:blonde, :red]
95 # HairColors.class_variable_get("@@hair_colors") # => [:blonde, :red]
96 #
97 # If you want to opt out the instance writer method, pass
98 # <tt>instance_writer: false</tt> or <tt>instance_accessor: false</tt>.
99 #
100 # module HairColors
101 # mattr_writer :hair_colors, instance_writer: false
102 # end
103 #
104 # class Person
105 # include HairColors
106 # end
107 #
108 # Person.new.hair_colors = [:blonde, :red] # => NoMethodError
109 #
110 # You can set a default value for the attribute.
111 #
112 # module HairColors
113 # mattr_writer :hair_colors, default: [:brown, :black, :blonde, :red]
114 # end
115 #
116 # class Person
117 # include HairColors
118 # end
119 #
120 # Person.class_variable_get("@@hair_colors") # => [:brown, :black, :blonde, :red]
121 def mattr_writer(*syms, instance_writer: true, instance_accessor: true, default: nil)
122 syms.each do |sym|
123 raise NameError.new("invalid attribute name: #{sym}") unless /\A[_A-Za-z]\w*\z/.match?(sym)
124 class_eval(<<-EOS, __FILE__, __LINE__ + 1)
125 @@#{sym} = nil unless defined? @@#{sym}
126
127 def self.#{sym}=(obj)
128 @@#{sym} = obj
129 end
130 EOS
131
132 if instance_writer && instance_accessor
133 class_eval(<<-EOS, __FILE__, __LINE__ + 1)
134 def #{sym}=(obj)
135 @@#{sym} = obj
136 end
137 EOS
138 end
139
140 sym_default_value = (block_given? && default.nil?) ? yield : default
141 send("#{sym}=", sym_default_value) unless sym_default_value.nil?
142 end
143 end

146 # Defines both class and instance accessors for class attributes.
147 # All class and instance methods created will be public, even if
148 # this method is called with a private or protected access modifier.
149 #
150 # module HairColors
151 # mattr_accessor :hair_colors
152 # end
153 #
154 # class Person
155 # include HairColors
156 # end
157 #
158 # HairColors.hair_colors = [:brown, :black, :blonde, :red]
159 # HairColors.hair_colors # => [:brown, :black, :blonde, :red]
160 # Person.new.hair_colors # => [:brown, :black, :blonde, :red]
161 #
162 # If a subclass changes the value then that would also change the value for
163 # parent class. Similarly if parent class changes the value then that would
164 # change the value of subclasses too.
165 #
166 # class Male < Person
167 # end
168 #
169 # Male.new.hair_colors << :blue
170 # Person.new.hair_colors # => [:brown, :black, :blonde, :red, :blue]
171 #
172 # To opt out of the instance writer method, pass <tt>instance_writer: false</tt>.
173 # To opt out of the instance reader method, pass <tt>instance_reader: false</tt>.
174 #
175 # module HairColors
176 # mattr_accessor :hair_colors, instance_writer: false, instance_reader: false
177 # end
178 #
179 # class Person
180 # include HairColors
181 # end
182 #
183 # Person.new.hair_colors = [:brown] # => NoMethodError
184 # Person.new.hair_colors # => NoMethodError
185 #
186 # Or pass <tt>instance_accessor: false</tt>, to opt out both instance methods.
187 #
188 # module HairColors
189 # mattr_accessor :hair_colors, instance_accessor: false
190 # end
191 #
192 # class Person
193 # include HairColors
194 # end
195 #
196 # Person.new.hair_colors = [:brown] # => NoMethodError
197 # Person.new.hair_colors # => NoMethodError
198 #
199 # You can set a default value for the attribute.
200 #
201 # module HairColors
202 # mattr_accessor :hair_colors, default: [:brown, :black, :blonde, :red]
203 # end
204 #
205 # class Person
206 # include HairColors
207 # end
208 #
209 # Person.class_variable_get("@@hair_colors") # => [:brown, :black, :blonde, :red]
210 def mattr_accessor(*syms, instance_reader: true, instance_writer: true, instance_accessor: true, default: nil, &blk)
211 mattr_reader(*syms, instance_reader: instance_reader, instance_accessor: instance_accessor, default: default, &blk)
212 mattr_writer(*syms, instance_writer: instance_writer, instance_accessor: instance_accessor, default: default)
213 end
214 alias :cattr_accessor :mattr_accessor
