7 # Converts a string to a Time value.
8 # The +form+ can be either :utc or :local (default :local).
9 #
10 # The time is parsed using Time.parse method.
11 # If +form+ is :local, then the time is in the system timezone.
12 # If the date part is missing then the current date is used and if
13 # the time part is missing then it is assumed to be 00:00:00.
14 #
15 # "13-12-2012".to_time # => 2012-12-13 00:00:00 +0100
16 # "06:12".to_time # => 2012-12-13 06:12:00 +0100
17 # "2012-12-13 06:12".to_time # => 2012-12-13 06:12:00 +0100
18 # "2012-12-13T06:12".to_time # => 2012-12-13 06:12:00 +0100
19 # "2012-12-13T06:12".to_time(:utc) # => 2012-12-13 06:12:00 UTC
20 # "12/13/2012".to_time # => ArgumentError: argument out of range
21 def to_time(form = :local)
22 parts = Date._parse(self, false)
23 used_keys = %i(year mon mday hour min sec sec_fraction offset)
24 return if (parts.keys & used_keys).empty?
25
26 now = Time.now
27 time = Time.new(
28 parts.fetch(:year, now.year),
29 parts.fetch(:mon, now.month),
30 parts.fetch(:mday, now.day),
31 parts.fetch(:hour, 0),
32 parts.fetch(:min, 0),
33 parts.fetch(:sec, 0) + parts.fetch(:sec_fraction, 0),
34 parts.fetch(:offset, form == :utc ? 0 : nil)
35 )
36
37 form == :utc ? time.utc : time.to_time
38 end
