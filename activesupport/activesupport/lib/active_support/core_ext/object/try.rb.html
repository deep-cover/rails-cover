28 ##
29 # :method: try
30 #
31 # :call-seq:
32 # try(*a, &b)
33 #
34 # Invokes the public method whose name goes as first argument just like
35 # +public_send+ does, except that if the receiver does not respond to it the
36 # call returns +nil+ rather than raising an exception.
37 #
38 # This method is defined to be able to write
39 #
40 # @person.try(:name)
41 #
42 # instead of
43 #
44 # @person.name if @person
45 #
46 # +try+ calls can be chained:
47 #
48 # @person.try(:spouse).try(:name)
49 #
50 # instead of
51 #
52 # @person.spouse.name if @person && @person.spouse
53 #
54 # +try+ will also return +nil+ if the receiver does not respond to the method:
55 #
56 # @person.try(:non_existing_method) # => nil
57 #
58 # instead of
59 #
60 # @person.non_existing_method if @person.respond_to?(:non_existing_method) # => nil
61 #
62 # +try+ returns +nil+ when called on +nil+ regardless of whether it responds
63 # to the method:
64 #
65 # nil.try(:to_i) # => nil, rather than 0
66 #
67 # Arguments and blocks are forwarded to the method if invoked:
68 #
69 # @posts.try(:each_slice, 2) do |a, b|
70 # ...
71 # end
72 #
73 # The number of arguments in the signature must match. If the object responds
74 # to the method the call is attempted and +ArgumentError+ is still raised
75 # in case of argument mismatch.
76 #
77 # If +try+ is called without arguments it yields the receiver to a given
78 # block unless it is +nil+:
79 #
80 # @person.try do |p|
81 # ...
82 # end
83 #
84 # You can also call try with a block without accepting an argument, and the block
85 # will be instance_eval'ed instead:
86 #
87 # @person.try { upcase.truncate(50) }
88 #
89 # Please also note that +try+ is defined on +Object+. Therefore, it won't work
90 # with instances of classes that do not have +Object+ among their ancestors,
91 # like direct subclasses of +BasicObject+.
92
93 ##
94 # :method: try!
95 #
96 # :call-seq:
97 # try!(*a, &b)
98 #
99 # Same as #try, but raises a +NoMethodError+ exception if the receiver is
100 # not +nil+ and does not implement the tried method.
101 #
102 # "a".try!(:upcase) # => "A"
103 # nil.try!(:upcase) # => nil
104 # 123.try!(:upcase) # => NoMethodError: undefined method `upcase' for 123:Integer
105end
