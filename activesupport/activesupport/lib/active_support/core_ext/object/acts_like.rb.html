3class Object
4 # A duck-type assistant method. For example, Active Support extends Date
5 # to define an <tt>acts_like_date?</tt> method, and extends Time to define
6 # <tt>acts_like_time?</tt>. As a result, we can do <tt>x.acts_like?(:time)</tt> and
7 # <tt>x.acts_like?(:date)</tt> to do duck-type-safe comparisons, since classes that
8 # we want to act like Time simply need to define an <tt>acts_like_time?</tt> method.
9 def acts_like?(duck)
10 case duck
11 when :time
12 respond_to? :acts_like_time?
13 when :date
14 respond_to? :acts_like_date?
15 when :string
16 respond_to? :acts_like_string?
17 else
18 respond_to? :"acts_like_#{duck}?"
19 end
20 end
21end
