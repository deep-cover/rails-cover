4 # Returns true if this object is included in the argument. Argument must be
5 # any object which responds to +#include?+. Usage:
6 #
7 # characters = ["Konata", "Kagami", "Tsukasa"]
8 # "Konata".in?(characters) # => true
9 #
10 # This will throw an +ArgumentError+ if the argument doesn't respond
11 # to +#include?+.
12 def in?(another_object)
13 another_object.include?(self)
14 rescue NoMethodError
15 raise ArgumentError.new("The parameter passed to #in? must respond to #include?")
16 end

18 # Returns the receiver if it's included in the argument otherwise returns +nil+.
19 # Argument must be any object which responds to +#include?+. Usage:
20 #
21 # params[:bucket_type].presence_in %w( project calendar )
22 #
23 # This will throw an +ArgumentError+ if the argument doesn't respond to +#include?+.
24 #

27 in?(another_object) ? self : nil
