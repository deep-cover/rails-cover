110 # Provides precise Date calculations for years, months, and days. The +options+ parameter takes a hash with
111 # any of these keys: <tt>:years</tt>, <tt>:months</tt>, <tt>:weeks</tt>, <tt>:days</tt>.
112 def advance(options)
113 options = options.dup
114 d = self
115 d = d >> options.delete(:years) * 12 if options[:years]
116 d = d >> options.delete(:months) if options[:months]
117 d = d + options.delete(:weeks) * 7 if options[:weeks]
118 d = d + options.delete(:days) if options[:days]
119 d
120 end
