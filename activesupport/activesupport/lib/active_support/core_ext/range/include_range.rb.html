5 # Extends the default Range#include? to support range comparisons.
6 # (1..5).include?(1..5) # => true
7 # (1..5).include?(2..3) # => true
8 # (1..5).include?(2..6) # => false
9 #
10 # The native Range#include? behavior is untouched.
11 # ('a'..'f').include?('c') # => true
12 # (5..9).include?(11) # => false
13 def include?(value)
14 if value.is_a?(::Range)
15 # 1...10 includes 1..9 but it does not include 1..10.
16 operator = exclude_end? && !value.exclude_end? ? :< : :<=
17 super(value.first) && value.last.send(operator, last)
18 else
19 super
20 end
21 end
22 end
