6module ActiveSupport
7 # ActiveSupport::Subscriber is an object set to consume
8 # ActiveSupport::Notifications. The subscriber dispatches notifications to
9 # a registered object based on its given namespace.
10 #
11 # An example would be an Active Record subscriber responsible for collecting
12 # statistics about queries:
13 #
14 # module ActiveRecord
15 # class StatsSubscriber < ActiveSupport::Subscriber
16 # attach_to :active_record
17 #
18 # def sql(event)
19 # Statsd.timing("sql.#{event.payload[:name]}", event.duration)
20 # end
21 # end
22 # end
23 #
24 # After configured, whenever a "sql.active_record" notification is published,
25 # it will properly dispatch the event (ActiveSupport::Notifications::Event) to
26 # the +sql+ method.
27 class Subscriber
28 class << self
29 # Attach the subscriber to a namespace.
30 def attach_to(namespace, subscriber = new, notifier = ActiveSupport::Notifications)
31 @namespace = namespace
32 @subscriber = subscriber
33 @notifier = notifier
34
35 subscribers << subscriber
36
37 # Add event subscribers for all existing methods on the class.
38 subscriber.public_methods(false).each do |event|
39 add_event_subscriber(event)
40 end
41 end
42
43 # Adds event subscribers for all new methods added to the class.
44 def method_added(event)
45 # Only public methods are added as subscribers, and only if a notifier
46 # has been set up. This means that subscribers will only be set up for
47 # classes that call #attach_to.
48 if public_method_defined?(event) && notifier
49 add_event_subscriber(event)
50 end
51 end
52
53 def subscribers
54 @@subscribers ||= []
55 end

65 def add_event_subscriber(event) # :doc:
66 return if %w{ start finish }.include?(event.to_s)
67
68 pattern = "#{event}.#{namespace}"
69
70 # Don't add multiple subscribers (eg. if methods are redefined).
71 return if subscriber.patterns.include?(pattern)
72
73 subscriber.patterns << pattern
74 notifier.subscribe(pattern, subscriber)
75 end
76 end

78 attr_reader :patterns # :nodoc:

80 def initialize
81 @queue_key = [self.class.name, object_id].join "-"
82 @patterns = []

86 def start(name, id, payload)
87 e = ActiveSupport::Notifications::Event.new(name, Time.now, nil, id, payload)
88 parent = event_stack.last
89 parent << e if parent
90
91 event_stack.push e
92 end
93
94 def finish(name, id, payload)
95 finished = Time.now
96 event = event_stack.pop
97 event.end = finished
98 event.payload.merge!(payload)
99
100 method = name.split(".".freeze).first
101 send(method, event)
102 end
