121 def to_tag(key, value, options)
122 type_name = options.delete(:type)
123 merged_options = options.merge(root: key, skip_instruct: true)
124
125 if value.is_a?(::Method) || value.is_a?(::Proc)
126 if value.arity == 1
127 value.call(merged_options)
128 else
129 value.call(merged_options, key.to_s.singularize)
130 end
131 elsif value.respond_to?(:to_xml)
132 value.to_xml(merged_options)
133 else
134 type_name ||= TYPE_NAMES[value.class.name]
135 type_name ||= value.class.name if value && !value.respond_to?(:to_str)
136 type_name = type_name.to_s if type_name
137 type_name = "dateTime" if type_name == "datetime"
138
139 key = rename_key(key.to_s, options)
140
141 attributes = options[:skip_types] || type_name.nil? ? {} : { type: type_name }
142 attributes[:nil] = true if value.nil?
143
144 encoding = options[:encoding] || DEFAULT_ENCODINGS[type_name]
145 attributes[:encoding] = encoding if encoding
146
147 formatted_value = FORMATTING[type_name] && !value.nil? ?
148 FORMATTING[type_name].call(value) : value
149
150 options[:builder].tag!(key, formatted_value, attributes)
151 end
152 end

154 def rename_key(key, options = {})
155 camelize = options[:camelize]
156 dasherize = !options.has_key?(:dasherize) || options[:dasherize]
157 if camelize
158 key = true == camelize ? key.camelize : key.camelize(camelize)
159 end
160 key = _dasherize(key) if dasherize
161 key
162 end
163
164 private
165
166 def _dasherize(key)
167 # $2 must be a non-greedy regex for this to work
168 left, middle, right = /\A(_*)(.*?)(_*)\Z/.match(key.strip)[1, 3]
169 "#{left}#{middle.tr('_ ', '--')}#{right}"
170 end
