8 # = Notifications
9 #
10 # <tt>ActiveSupport::Notifications</tt> provides an instrumentation API for
11 # Ruby.
12 #
13 # == Instrumenters
14 #
15 # To instrument an event you just need to do:
16 #
17 # ActiveSupport::Notifications.instrument('render', extra: :information) do
18 # render plain: 'Foo'
19 # end
20 #
21 # That first executes the block and then notifies all subscribers once done.
22 #
23 # In the example above +render+ is the name of the event, and the rest is called
24 # the _payload_. The payload is a mechanism that allows instrumenters to pass
25 # extra information to subscribers. Payloads consist of a hash whose contents
26 # are arbitrary and generally depend on the event.
27 #
28 # == Subscribers
29 #
30 # You can consume those events and the information they provide by registering
31 # a subscriber.
32 #
33 # ActiveSupport::Notifications.subscribe('render') do |name, start, finish, id, payload|
34 # name # => String, name of the event (such as 'render' from above)
35 # start # => Time, when the instrumented block started execution
36 # finish # => Time, when the instrumented block ended execution
37 # id # => String, unique ID for this notification
38 # payload # => Hash, the payload
39 # end
40 #
41 # For instance, let's store all "render" events in an array:
42 #
43 # events = []
44 #
45 # ActiveSupport::Notifications.subscribe('render') do |*args|
46 # events << ActiveSupport::Notifications::Event.new(*args)
47 # end
48 #
49 # That code returns right away, you are just subscribing to "render" events.
50 # The block is saved and will be called whenever someone instruments "render":
51 #
52 # ActiveSupport::Notifications.instrument('render', extra: :information) do
53 # render plain: 'Foo'
54 # end
55 #
56 # event = events.first
57 # event.name # => "render"
58 # event.duration # => 10 (in milliseconds)
59 # event.payload # => { extra: :information }
60 #
61 # The block in the <tt>subscribe</tt> call gets the name of the event, start
62 # timestamp, end timestamp, a string with a unique identifier for that event
63 # (something like "535801666f04d0298cd6"), and a hash with the payload, in
64 # that order.
65 #
66 # If an exception happens during that particular instrumentation the payload will
67 # have a key <tt>:exception</tt> with an array of two elements as value: a string with
68 # the name of the exception class, and the exception message.
69 # The <tt>:exception_object</tt> key of the payload will have the exception
70 # itself as the value.
71 #
72 # As the previous example depicts, the class <tt>ActiveSupport::Notifications::Event</tt>
73 # is able to take the arguments as they come and provide an object-oriented
74 # interface to that data.
75 #
76 # It is also possible to pass an object which responds to <tt>call</tt> method
77 # as the second parameter to the <tt>subscribe</tt> method instead of a block:
78 #
79 # module ActionController
80 # class PageRequest
81 # def call(name, started, finished, unique_id, payload)
82 # Rails.logger.debug ['notification:', name, started, finished, unique_id, payload].join(' ')
83 # end
84 # end
85 # end
86 #
87 # ActiveSupport::Notifications.subscribe('process_action.action_controller', ActionController::PageRequest.new)
88 #
89 # resulting in the following output within the logs including a hash with the payload:
90 #
91 # notification: process_action.action_controller 2012-04-13 01:08:35 +0300 2012-04-13 01:08:35 +0300 af358ed7fab884532ec7 {
92 # controller: "Devise::SessionsController",
93 # action: "new",
94 # params: {"action"=>"new", "controller"=>"devise/sessions"},
95 # format: :html,
96 # method: "GET",
97 # path: "/login/sign_in",
98 # status: 200,
99 # view_runtime: 279.3080806732178,
100 # db_runtime: 40.053
101 # }
102 #
103 # You can also subscribe to all events whose name matches a certain regexp:
104 #
105 # ActiveSupport::Notifications.subscribe(/render/) do |*args|
106 # ...
107 # end
108 #
109 # and even pass no argument to <tt>subscribe</tt>, in which case you are subscribing
110 # to all events.
111 #
112 # == Temporary Subscriptions
113 #
114 # Sometimes you do not want to subscribe to an event for the entire life of
115 # the application. There are two ways to unsubscribe.
116 #
117 # WARNING: The instrumentation framework is designed for long-running subscribers,
118 # use this feature sparingly because it wipes some internal caches and that has
119 # a negative impact on performance.
120 #
121 # === Subscribe While a Block Runs
122 #
123 # You can subscribe to some event temporarily while some block runs. For
124 # example, in
125 #
126 # callback = lambda {|*args| ... }
127 # ActiveSupport::Notifications.subscribed(callback, "sql.active_record") do
128 # ...
129 # end
130 #
131 # the callback will be called for all "sql.active_record" events instrumented
132 # during the execution of the block. The callback is unsubscribed automatically
133 # after that.
134 #
135 # === Manual Unsubscription
136 #
137 # The +subscribe+ method returns a subscriber object:
138 #
139 # subscriber = ActiveSupport::Notifications.subscribe("render") do |*args|
140 # ...
141 # end
142 #
143 # To prevent that block from being called anymore, just unsubscribe passing
144 # that reference:
145 #
146 # ActiveSupport::Notifications.unsubscribe(subscriber)
147 #
148 # You can also unsubscribe by passing the name of the subscriber object. Note
149 # that this will unsubscribe all subscriptions with the given name:
150 #
151 # ActiveSupport::Notifications.unsubscribe("render")
152 #
153 # == Default Queue
154 #
155 # Notifications ships with a queue implementation that consumes and publishes events
156 # to all log subscribers. You can use any queue implementation you want.
157 #
158 module Notifications
159 class << self
160 attr_accessor :notifier
161
162 def publish(name, *args)
163 notifier.publish(name, *args)
164 end
165
166 def instrument(name, payload = {})
167 if notifier.listening?(name)
168 instrumenter.instrument(name, payload) { yield payload if block_given? }
169 else
170 yield payload if block_given?
171 end
172 end
173
174 def subscribe(*args, &block)
175 notifier.subscribe(*args, &block)
176 end
177
178 def subscribed(callback, *args, &block)
179 subscriber = subscribe(*args, &callback)
180 yield
181 ensure
182 unsubscribe(subscriber)
183 end
184
185 def unsubscribe(subscriber_or_name)
186 notifier.unsubscribe(subscriber_or_name)
187 end
188
189 def instrumenter
190 InstrumentationRegistry.instance.instrumenter_for(notifier)
191 end
192 end

194 # This class is a registry which holds all of the +Instrumenter+ objects
195 # in a particular thread local. To access the +Instrumenter+ object for a
196 # particular +notifier+, you can call the following method:
197 #
198 # InstrumentationRegistry.instrumenter_for(notifier)
199 #
200 # The instrumenters for multiple notifiers are held in a single instance of
201 # this class.
202 class InstrumentationRegistry # :nodoc:
203 extend ActiveSupport::PerThreadRegistry
204
205 def initialize
206 @registry = {}
207 end
208
209 def instrumenter_for(notifier)
210 @registry[notifier] ||= Instrumenter.new(notifier)
211 end
212 end
213
214 self.notifier = Fanout.new
215 end
