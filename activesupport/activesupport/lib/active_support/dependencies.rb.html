82 # The WatchStack keeps a stack of the modules being watched as files are
83 # loaded. If a file in the process of being loaded (parent.rb) triggers the
84 # load of another file (child.rb) the stack will ensure that child.rb
85 # handles the new constants.
86 #
87 # If child.rb is being autoloaded, its constants will be added to
88 # autoloaded_constants. If it was being required, they will be discarded.
89 #
90 # This is handled by walking back up the watch stack and adding the constants
91 # found by child.rb to the list of original constants in parent.rb.
92 class WatchStack
93 include Enumerable
94
95 # @watching is a stack of lists of constants being watched. For instance,
96 # if parent.rb is autoloaded, the stack will look like [[Object]]. If
97 # parent.rb then requires namespace/child.rb, the stack will look like
98 # [[Object], [Namespace]].
99
100 def initialize
101 @watching = []
102 @stack = Hash.new { |h, k| h[k] = [] }
103 end
104
105 def each(&block)
106 @stack.each(&block)
107 end

109 def watching?

113 # Returns a list of new constants found since the last call to
114 # <tt>watch_namespaces</tt>.
115 def new_constants
116 constants = []
117
118 # Grab the list of namespaces that we're looking for new constants under
119 @watching.last.each do |namespace|
120 # Retrieve the constants that were present under the namespace when watch_namespaces
121 # was originally called
122 original_constants = @stack[namespace].last
123
124 mod = Inflector.constantize(namespace) if Dependencies.qualified_const_defined?(namespace)
125 next unless mod.is_a?(Module)
126
127 # Get a list of the constants that were added
128 new_constants = mod.constants(false) - original_constants
129
130 # @stack[namespace] returns an Array of the constants that are being evaluated
131 # for that namespace. For instance, if parent.rb requires child.rb, the first
132 # element of @stack[Object] will be an Array of the constants that were present
133 # before parent.rb was required. The second element will be an Array of the
134 # constants that were present before child.rb was required.
135 @stack[namespace].each do |namespace_constants|
136 namespace_constants.concat(new_constants)
137 end
138
139 # Normalize the list of new constants, and add them to the list we will return
140 new_constants.each do |suffix|
141 constants << ([namespace, suffix] - ["Object"]).join("::".freeze)
142 end
143 end
144 constants
145 ensure
146 # A call to new_constants is always called after a call to watch_namespaces
147 pop_modules(@watching.pop)
148 end
149
150 # Add a set of modules to the watch stack, remembering the initial
151 # constants.
152 def watch_namespaces(namespaces)
153 @watching << namespaces.map do |namespace|
154 module_name = Dependencies.to_constant_name(namespace)
155 original_constants = Dependencies.qualified_const_defined?(module_name) ?
156 Inflector.constantize(module_name).constants(false) : []
157
158 @stack[module_name] << original_constants
159 module_name
160 end
161 end

164 def pop_modules(modules)
165 modules.each { |mod| @stack[mod].pop }
166 end

343 def require_or_load(file_name, const_path = nil)
344 file_name = $` if file_name =~ /\.rb\z/
345 expanded = File.expand_path(file_name)
346 return if loaded.include?(expanded)
347
348 Dependencies.load_interlock do
349 # Maybe it got loaded while we were waiting for our lock:
350 return if loaded.include?(expanded)
351
352 # Record that we've seen this file *before* loading it to avoid an
353 # infinite loop with mutual dependencies.
354 loaded << expanded
355 loading << expanded
356
357 begin
358 if load?
359 # Enable warnings if this file has not been loaded before and
360 # warnings_on_first_load is set.
361 load_args = ["#{file_name}.rb"]
362 load_args << const_path unless const_path.nil?
363
364 if !warnings_on_first_load || history.include?(expanded)
365 result = load_file(*load_args)
366 else
367 enable_warnings { result = load_file(*load_args) }
368 end
369 else
370 result = require file_name
371 end
372 rescue Exception
373 loaded.delete expanded
374 raise
375 ensure
376 loading.pop
377 end
378
379 # Record history *after* loading so first load gets warnings.
380 history << expanded
381 result
382 end
383 end

390 # Given +path+, a filesystem path to a ruby file, return an array of
391 # constant paths which would cause Dependencies to attempt to load this

394 path = $` if path =~ /\.rb\z/
395 expanded_path = File.expand_path(path)

440 # Attempt to autoload the provided module name by searching for a directory
441 # matching the expected path suffix. If found, the module is created and
442 # assigned to +into+'s constants with the name +const_name+. Provided that
443 # the directory was loaded from a reloadable base path, it is added to the
444 # set of constants that are to be unloaded.
445 def autoload_module!(into, const_name, qualified_name, path_suffix)
446 return nil unless base_path = autoloadable_module?(path_suffix)
447 mod = Module.new
448 into.const_set const_name, mod
449 autoloaded_constants << qualified_name unless autoload_once_paths.include?(base_path)
450 mod
451 end

453 # Load the file at the provided path. +const_paths+ is a set of qualified
454 # constant names. When loading the file, Dependencies will watch for the
455 # addition of these constants. Each that is defined will be marked as
456 # autoloaded, and will be removed when Dependencies.clear is next called.
457 #
458 # If the second parameter is left off, then Dependencies will construct a
459 # set of names that the file at +path+ may define. See
460 # +loadable_constants_for_path+ for more details.
461 def load_file(path, const_paths = loadable_constants_for_path(path))
462 const_paths = [const_paths].compact unless const_paths.is_a? Array
463 parent_paths = const_paths.collect { |const_path| const_path[/.*(?=::)/] || ::Object }
464
465 result = nil
466 newly_defined_paths = new_constants_in(*parent_paths) do
467 result = Kernel.load path
468 end
469
470 autoloaded_constants.concat newly_defined_paths unless load_once_path?(path)
471 autoloaded_constants.uniq!
472 result
473 end

481 # Load the constant named +const_name+ which is missing from +from_mod+. If
482 # it is not possible to load the constant into from_mod, try its parent
483 # module using +const_missing+.
484 def load_missing_constant(from_mod, const_name)
485 unless qualified_const_defined?(from_mod.name) && Inflector.constantize(from_mod.name).equal?(from_mod)
486 raise ArgumentError, "A copy of #{from_mod} has been removed from the module tree but is still active!"
487 end
488
489 qualified_name = qualified_name_for from_mod, const_name
490 path_suffix = qualified_name.underscore
491
492 file_path = search_for_file(path_suffix)
493
494 if file_path
495 expanded = File.expand_path(file_path)
496 expanded.sub!(/\.rb\z/, "".freeze)
497
498 if loading.include?(expanded)
499 raise "Circular dependency detected while autoloading constant #{qualified_name}"
500 else
501 require_or_load(expanded, qualified_name)
502 raise LoadError, "Unable to autoload constant #{qualified_name}, expected #{file_path} to define it" unless from_mod.const_defined?(const_name, false)
503 return from_mod.const_get(const_name)
504 end
505 elsif mod = autoload_module!(from_mod, const_name, qualified_name, path_suffix)
506 return mod
507 elsif (parent = from_mod.parent) && parent != from_mod &&
508 ! from_mod.parents.any? { |p| p.const_defined?(const_name, false) }
509 # If our parents do not have a constant named +const_name+ then we are free
510 # to attempt to load upwards. If they do have such a constant, then this
511 # const_missing must be due to from_mod::const_name, which should not
512 # return constants from from_mod's parents.
513 begin
514 # Since Ruby does not pass the nesting at the point the unknown
515 # constant triggered the callback we cannot fully emulate constant
516 # name lookup and need to make a trade-off: we are going to assume
517 # that the nesting in the body of Foo::Bar is [Foo::Bar, Foo] even
518 # though it might not be. Counterexamples are
519 #
520 # class Foo::Bar
521 # Module.nesting # => [Foo::Bar]
522 # end
523 #
524 # or
525 #
526 # module M::N
527 # module S::T
528 # Module.nesting # => [S::T, M::N]
529 # end
530 # end
531 #
532 # for example.
533 return parent.const_missing(const_name)
534 rescue NameError => e
535 raise unless e.missing_name? qualified_name_for(parent, const_name)
536 end
537 end

639 # Run the provided block and detect the new constants that were loaded during
640 # its execution. Constants may only be regarded as 'new' once -- so if the
641 # block calls +new_constants_in+ again, then the constants defined within the
642 # inner call will not be reported in this one.
643 #
644 # If the provided block does not run to completion, and instead raises an
645 # exception, any new constants are regarded as being only partially defined
646 # and will be removed immediately.
647 def new_constants_in(*descs)
648 constant_watch_stack.watch_namespaces(descs)
649 success = false
650
651 begin
652 yield # Now yield to the code that is to define new constants.
653 success = true
654 ensure
655 new_constants = constant_watch_stack.new_constants
656
657 return new_constants if success
658
659 # Remove partially loaded constants.
660 new_constants.each { |c| remove_constant(c) }
661 end

664 # Convert the provided const desc to a qualified constant name (as a string).
665 # A module, class, symbol, or string may be provided.

667 case desc
668 when String then desc.sub(/^::/, "")
669 when Symbol then desc.to_s
670 when Module
671 desc.name ||
672 raise(ArgumentError, "Anonymous modules have no name to be referenced by")
673 else raise TypeError, "Not a valid constant descriptor: #{desc.inspect}"

677 def remove_constant(const) #:nodoc:
678 # Normalize ::Foo, ::Object::Foo, Object::Foo, Object::Object::Foo, etc. as Foo.
679 normalized = const.to_s.sub(/\A::/, "")
680 normalized.sub!(/\A(Object::)+/, "")
681
682 constants = normalized.split("::")
683 to_remove = constants.pop
684
685 # Remove the file path from the loaded list.
686 file_path = search_for_file(const.underscore)
687 if file_path
688 expanded = File.expand_path(file_path)
689 expanded.sub!(/\.rb\z/, "")
690 loaded.delete(expanded)
691 end
692
693 if constants.empty?
694 parent = Object
695 else
696 # This method is robust to non-reachable constants.
697 #
698 # Non-reachable constants may be passed if some of the parents were
699 # autoloaded and already removed. It is easier to do a sanity check
700 # here than require the caller to be clever. We check the parent
701 # rather than the very const argument because we do not want to
702 # trigger Kernel#autoloads, see the comment below.
703 parent_name = constants.join("::")
704 return unless qualified_const_defined?(parent_name)
705 parent = constantize(parent_name)
706 end

708 # In an autoloaded user.rb like this
709 #
710 # autoload :Foo, 'foo'
711 #
712 # class User < ActiveRecord::Base
713 # end
714 #
715 # we correctly register "Foo" as being autoloaded. But if the app does
716 # not use the "Foo" constant we need to be careful not to trigger
717 # loading "foo.rb" ourselves. While #const_defined? and #const_get? do
718 # require the file, #autoload? and #remove_const don't.
719 #
720 # We are going to remove the constant nonetheless ---which exists as
721 # far as Ruby is concerned--- because if the user removes the macro
722 # call from a class or module that were not autoloaded, as in the
723 # example above with Object, accessing to that constant must err.
724 unless parent.autoload?(to_remove)
725 begin
726 constantized = parent.const_get(to_remove, false)
727 rescue NameError
728 # The constant is no longer reachable, just skip it.
729 return
