11 # Chars enables you to work transparently with UTF-8 encoding in the Ruby
12 # String class without having extensive knowledge about the encoding. A
13 # Chars object accepts a string upon initialization and proxies String
14 # methods in an encoding safe manner. All the normal String methods are also
15 # implemented on the proxy.
16 #
17 # String methods are proxied through the Chars object, and can be accessed
18 # through the +mb_chars+ method. Methods which would normally return a
19 # String object now return a Chars object so methods can be chained.
20 #
21 # 'The Perfect String '.mb_chars.downcase.strip.normalize
22 # # => #<ActiveSupport::Multibyte::Chars:0x007fdc434ccc10 @wrapped_string="the perfect string">
23 #
24 # Chars objects are perfectly interchangeable with String objects as long as
25 # no explicit class checks are made. If certain methods do explicitly check
26 # the class, call +to_s+ before you pass chars objects to them.
27 #
28 # bad.explicit_checking_method 'T'.mb_chars.downcase.to_s
29 #
30 # The default Chars implementation assumes that the encoding of the string
31 # is UTF-8, if you want to handle different encodings you can write your own
32 # multibyte string handler and configure it through
33 # ActiveSupport::Multibyte.proxy_class.
34 #
35 # class CharsForUTF32
36 # def size
37 # @wrapped_string.size / 4
38 # end
39 #
40 # def self.accepts?(string)
41 # string.length % 4 == 0
42 # end
43 # end
44 #
45 # ActiveSupport::Multibyte.proxy_class = CharsForUTF32
46 class Chars
47 include Comparable

54 # Creates a new Chars instance by wrapping _string_.
55 def initialize(string)
56 @wrapped_string = string

60 # Forward all undefined methods to the wrapped string.
61 def method_missing(method, *args, &block)
62 result = @wrapped_string.__send__(method, *args, &block)
63 if /!$/.match?(method)
64 self if result
65 else
66 result.kind_of?(String) ? chars(result) : result
67 end
68 end

83 # Works just like <tt>String#split</tt>, with the exception that the items
84 # in the resulting list are Chars instances instead of String. This makes
85 # chaining methods easier.
86 #
87 # 'Café périferôl'.mb_chars.split(/é/).map { |part| part.upcase.to_s } # => ["CAF", " P", "RIFERÔL"]
88 def split(*args)
89 @wrapped_string.split(*args).map { |i| self.class.new(i) }
90 end

92 # Works like <tt>String#slice!</tt>, but returns an instance of
93 # Chars, or +nil+ if the string was not modified. The string will not be
94 # modified if the range given is out of bounds
95 #
96 # string = 'Welcome'
97 # string.mb_chars.slice!(3) # => #<ActiveSupport::Multibyte::Chars:0x000000038109b8 @wrapped_string="c">
98 # string # => 'Welome'
99 # string.mb_chars.slice!(0..3) # => #<ActiveSupport::Multibyte::Chars:0x00000002eb80a0 @wrapped_string="Welo">
100 # string # => 'me'
101 def slice!(*args)
102 string_sliced = @wrapped_string.slice!(*args)
103 if string_sliced
104 chars(string_sliced)
105 end
106 end

108 # Reverses all characters in the string.
109 #
110 # 'Café'.mb_chars.reverse.to_s # => 'éfaC'
111 def reverse
112 chars(Unicode.unpack_graphemes(@wrapped_string).reverse.flatten.pack("U*"))

115 # Limits the byte size of the string to a number of bytes without breaking
116 # characters. Usable when the storage for a string is limited for some
117 # reason.
118 #
119 # 'こんにちは'.mb_chars.limit(7).to_s # => "こん"
120 def limit(limit)
121 slice(0...translate_offset(limit))
122 end

145 # Converts the first character to uppercase and the remainder to lowercase.
146 #
147 # 'über'.mb_chars.capitalize.to_s # => "Über"
148 def capitalize
149 (slice(0) || chars("")).upcase + (slice(1..-1) || chars("")).downcase
150 end
151
152 # Capitalizes the first letter of every word, when possible.
153 #
154 # "ÉL QUE SE ENTERÓ".mb_chars.titleize.to_s # => "Él Que Se Enteró"
155 # "日本語".mb_chars.titleize.to_s # => "日本語"
156 def titleize
157 chars(downcase.to_s.gsub(/\b('?\S)/u) { Unicode.upcase($1) })
158 end

161 # Returns the KC normalization of the string by default. NFKC is
162 # considered the best normalization form for passing strings to databases
163 # and validations.
164 #
165 # * <tt>form</tt> - The form you want to normalize in. Should be one of the following:
166 # <tt>:c</tt>, <tt>:kc</tt>, <tt>:d</tt>, or <tt>:kd</tt>. Default is
167 # ActiveSupport::Multibyte::Unicode.default_normalization_form
168 def normalize(form = nil)
169 chars(Unicode.normalize(@wrapped_string, form))
170 end

172 # Performs canonical decomposition on all the characters.

174 # 'é'.length # => 2

218 def translate_offset(byte_offset)
219 return nil if byte_offset.nil?
220 return 0 if @wrapped_string == ""
221
222 begin
223 @wrapped_string.byteslice(0...byte_offset).unpack("U*").length
224 rescue ArgumentError
225 byte_offset -= 1
226 retry
227 end
228 end
