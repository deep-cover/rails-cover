7module ActiveSupport
8 # The TimeZone class serves as a wrapper around TZInfo::Timezone instances.
9 # It allows us to do the following:
10 #
11 # * Limit the set of zones provided by TZInfo to a meaningful subset of 134
12 # zones.
13 # * Retrieve and display zones with a friendlier name
14 # (e.g., "Eastern Time (US & Canada)" instead of "America/New_York").
15 # * Lazily load TZInfo::Timezone instances only when they're needed.
16 # * Create ActiveSupport::TimeWithZone instances via TimeZone's +local+,
17 # +parse+, +at+ and +now+ methods.
18 #
19 # If you set <tt>config.time_zone</tt> in the Rails Application, you can
20 # access this TimeZone object via <tt>Time.zone</tt>:
21 #
22 # # application.rb:
23 # class Application < Rails::Application
24 # config.time_zone = 'Eastern Time (US & Canada)'
25 # end
26 #
27 # Time.zone # => #<ActiveSupport::TimeZone:0x514834...>
28 # Time.zone.name # => "Eastern Time (US & Canada)"
29 # Time.zone.now # => Sun, 18 May 2008 14:30:44 EDT -04:00

193 # Assumes self represents an offset from UTC in seconds (as returned from
194 # Time#utc_offset) and turns this into an +HH:MM formatted string.
195 #
196 # ActiveSupport::TimeZone.seconds_to_utc_offset(-21_600) # => "-06:00"
197 def seconds_to_utc_offset(seconds, colon = true)
198 format = colon ? UTC_OFFSET_WITH_COLON : UTC_OFFSET_WITHOUT_COLON
199 sign = (seconds < 0 ? "-" : "+")
200 hours = seconds.abs / 3600
201 minutes = (seconds.abs % 3600) / 60
202 format % [sign, hours, minutes]
203 end

225 # Locate a specific time zone object. If the argument is a string, it
226 # is interpreted to mean the name of the timezone to locate. If it is a
227 # numeric value it is either the hour offset, or the second offset, of the
228 # timezone to find. (The first one with that offset will be returned.)
229 # Returns +nil+ if no such time zone is known to the system.
230 def [](arg)
231 case arg
232 when String
233 begin
234 @lazy_zones_map[arg] ||= create(arg)
235 rescue TZInfo::InvalidTimezoneIdentifier
236 nil
237 end
238 when Numeric, ActiveSupport::Duration
239 arg *= 3600 if arg.abs <= 13
240 all.find { |z| z.utc_offset == arg.to_i }
241 else
242 raise ArgumentError, "invalid argument to TimeZone[]: #{arg.inspect}"
243 end
244 end

358 # Method for creating new ActiveSupport::TimeWithZone instance in time zone
359 # of +self+ from an ISO 8601 string.
360 #
361 # Time.zone = 'Hawaii' # => "Hawaii"
362 # Time.zone.iso8601('1999-12-31T14:00:00') # => Fri, 31 Dec 1999 14:00:00 HST -10:00
363 #
364 # If the time components are missing then they will be set to zero.
365 #
366 # Time.zone = 'Hawaii' # => "Hawaii"
367 # Time.zone.iso8601('1999-12-31') # => Fri, 31 Dec 1999 00:00:00 HST -10:00
368 #
369 # If the string is invalid then an +ArgumentError+ will be raised unlike +parse+
370 # which usually returns +nil+ when given an invalid date string.
371 def iso8601(str)
372 parts = Date._iso8601(str)
373
374 raise ArgumentError, "invalid date" if parts.empty?
375
376 time = Time.new(
377 parts.fetch(:year),
378 parts.fetch(:mon),
379 parts.fetch(:mday),
380 parts.fetch(:hour, 0),
381 parts.fetch(:min, 0),
382 parts.fetch(:sec, 0) + parts.fetch(:sec_fraction, 0),
383 parts.fetch(:offset, 0)
384 )
385
386 if parts[:offset]
387 TimeWithZone.new(time.utc, self)
388 else
389 TimeWithZone.new(nil, self, time)
390 end
391 end

393 # Method for creating new ActiveSupport::TimeWithZone instance in time zone
394 # of +self+ from parsed string.
395 #
396 # Time.zone = 'Hawaii' # => "Hawaii"
397 # Time.zone.parse('1999-12-31 14:00:00') # => Fri, 31 Dec 1999 14:00:00 HST -10:00
398 #
399 # If upper components are missing from the string, they are supplied from
400 # TimeZone#now:
401 #
402 # Time.zone.now # => Fri, 31 Dec 1999 14:00:00 HST -10:00
403 # Time.zone.parse('22:30:00') # => Fri, 31 Dec 1999 22:30:00 HST -10:00
404 #
405 # However, if the date component is not provided, but any other upper
406 # components are supplied, then the day of the month defaults to 1:
407 #
408 # Time.zone.parse('Mar 2000') # => Wed, 01 Mar 2000 00:00:00 HST -10:00
409 #
410 # If the string is invalid then an +ArgumentError+ could be raised.
411 def parse(str, now = now())
412 parts_to_time(Date._parse(str, false), now)
413 end

415 # Method for creating new ActiveSupport::TimeWithZone instance in time zone
416 # of +self+ from an RFC 3339 string.
417 #
418 # Time.zone = 'Hawaii' # => "Hawaii"

445 # Parses +str+ according to +format+ and returns an ActiveSupport::TimeWithZone.
446 #
447 # Assumes that +str+ is a time in the time zone +self+,
448 # unless +format+ includes an explicit time zone.
449 # (This is the same behavior as +parse+.)
450 # In either case, the returned TimeWithZone has the timezone of +self+.
451 #
452 # Time.zone = 'Hawaii' # => "Hawaii"
453 # Time.zone.strptime('1999-12-31 14:00:00', '%Y-%m-%d %H:%M:%S') # => Fri, 31 Dec 1999 14:00:00 HST -10:00
454 #
455 # If upper components are missing from the string, they are supplied from
456 # TimeZone#now:
457 #
458 # Time.zone.now # => Fri, 31 Dec 1999 14:00:00 HST -10:00
459 # Time.zone.strptime('22:30:00', '%H:%M:%S') # => Fri, 31 Dec 1999 22:30:00 HST -10:00
460 #
461 # However, if the date component is not provided, but any other upper
462 # components are supplied, then the day of the month defaults to 1:
463 #
464 # Time.zone.strptime('Mar 2000', '%b %Y') # => Wed, 01 Mar 2000 00:00:00 HST -10:00
465 def strptime(str, format, now = now())
466 parts_to_time(DateTime._strptime(str, format), now)
467 end

532 def parts_to_time(parts, now)
533 raise ArgumentError, "invalid date" if parts.nil?
534 return if parts.empty?
535
536 if parts[:seconds]
537 time = Time.at(parts[:seconds])
538 else
539 time = Time.new(
540 parts.fetch(:year, now.year),
541 parts.fetch(:mon, now.month),
542 parts.fetch(:mday, parts[:year] || parts[:mon] ? 1 : now.day),
543 parts.fetch(:hour, 0),
544 parts.fetch(:min, 0),
545 parts.fetch(:sec, 0) + parts.fetch(:sec_fraction, 0),
546 parts.fetch(:offset, 0)
547 )
548 end
549
550 if parts[:offset] || parts[:seconds]
551 TimeWithZone.new(time.utc, self)
552 else
553 TimeWithZone.new(nil, self, time)
554 end
555 end
