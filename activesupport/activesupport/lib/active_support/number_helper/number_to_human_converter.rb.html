37 def determine_unit(units, exponent)
38 exp = DECIMAL_UNITS[exponent]

40 when Hash

42 when String, Symbol

49 def calculate_exponent(units)
50 exponent = number != 0 ? Math.log10(number.abs).floor : 0
51 unit_exponents(units).find { |e| exponent >= e } || 0
