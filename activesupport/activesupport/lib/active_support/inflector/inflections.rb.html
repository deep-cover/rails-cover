13 # A singleton instance of this class is yielded by Inflector.inflections,
14 # which can then be used to specify additional inflection rules. If passed
15 # an optional locale, rules for other languages can be specified. The
16 # default locale is <tt>:en</tt>. Only rules for English are provided.
17 #
18 # ActiveSupport::Inflector.inflections(:en) do |inflect|
19 # inflect.plural /^(ox)$/i, '\1\2en'
20 # inflect.singular /^(ox)en/i, '\1'
21 #
22 # inflect.irregular 'octopus', 'octopi'
23 #
24 # inflect.uncountable 'equipment'
25 # end
26 #
27 # New rules are added at the top. So in the example above, the irregular
28 # rule for octopus will now be the first of the pluralization and
29 # singularization rules that is runs. This guarantees that your rules run
30 # before any of the rules that may already have been loaded.
31 class Inflections
32 @__instance__ = Concurrent::Map.new

88 # Specifies a new acronym. An acronym must be specified as it will appear
89 # in a camelized string. An underscore string that contains the acronym
90 # will retain the acronym when passed to +camelize+, +humanize+, or
91 # +titleize+. A camelized string that contains the acronym will maintain
92 # the acronym when titleized or humanized, and will convert the acronym
93 # into a non-delimited single lowercase word when passed to +underscore+.
94 #
95 # acronym 'HTML'
96 # titleize 'html' # => 'HTML'
97 # camelize 'html' # => 'HTML'
98 # underscore 'MyHTML' # => 'my_html'
99 #
100 # The acronym, however, must occur as a delimited unit and not be part of
101 # another word for conversions to recognize it:
102 #
103 # acronym 'HTTP'
104 # camelize 'my_http_delimited' # => 'MyHTTPDelimited'
105 # camelize 'https' # => 'Https', not 'HTTPs'
106 # underscore 'HTTPS' # => 'http_s', not 'https'
107 #
108 # acronym 'HTTPS'
109 # camelize 'https' # => 'HTTPS'
110 # underscore 'HTTPS' # => 'https'
111 #
112 # Note: Acronyms that are passed to +pluralize+ will no longer be
113 # recognized, since the acronym will not occur as a delimited unit in the
114 # pluralized result. To work around this, you must specify the pluralized
115 # form as an acronym as well:
116 #
117 # acronym 'API'
118 # camelize(pluralize('api')) # => 'Apis'
119 #
120 # acronym 'APIs'
121 # camelize(pluralize('api')) # => 'APIs'
122 #
123 # +acronym+ may be used to specify any word that contains an acronym or
124 # otherwise needs to maintain a non-standard capitalization. The only
125 # restriction is that the word must begin with a capital letter.
126 #
127 # acronym 'RESTful'
128 # underscore 'RESTful' # => 'restful'
129 # underscore 'RESTfulController' # => 'restful_controller'
130 # titleize 'RESTfulController' # => 'RESTful Controller'
131 # camelize 'restful' # => 'RESTful'
132 # camelize 'restful_controller' # => 'RESTfulController'
133 #
134 # acronym 'McDonald'
135 # underscore 'McDonald' # => 'mcdonald'
136 # camelize 'mcdonald' # => 'McDonald'
137 def acronym(word)
138 @acronyms[word.downcase] = word
139 define_acronym_regex_patterns
140 end

142 # Specifies a new pluralization rule and its replacement. The rule can
143 # either be a string or a regular expression. The replacement should
144 # always be a string that may include references to the matched data from
145 # the rule.
146 def plural(rule, replacement)
147 @uncountables.delete(rule) if rule.is_a?(String)
148 @uncountables.delete(replacement)
149 @plurals.prepend([rule, replacement])
150 end

152 # Specifies a new singularization rule and its replacement. The rule can
153 # either be a string or a regular expression. The replacement should
154 # always be a string that may include references to the matched data from
155 # the rule.
156 def singular(rule, replacement)
157 @uncountables.delete(rule) if rule.is_a?(String)
158 @uncountables.delete(replacement)
159 @singulars.prepend([rule, replacement])
160 end

162 # Specifies a new irregular that applies to both pluralization and
163 # singularization at the same time. This can only be used for strings, not
164 # regular expressions. You simply pass the irregular in singular and
165 # plural form.
166 #
167 # irregular 'octopus', 'octopi'
168 # irregular 'person', 'people'
169 def irregular(singular, plural)
170 @uncountables.delete(singular)
171 @uncountables.delete(plural)
172
173 s0 = singular[0]
174 srest = singular[1..-1]
175
176 p0 = plural[0]
177 prest = plural[1..-1]
178
179 if s0.upcase == p0.upcase
180 plural(/(#{s0})#{srest}$/i, '\1' + prest)
181 plural(/(#{p0})#{prest}$/i, '\1' + prest)
182
183 singular(/(#{s0})#{srest}$/i, '\1' + srest)
184 singular(/(#{p0})#{prest}$/i, '\1' + srest)
185 else
186 plural(/#{s0.upcase}(?i)#{srest}$/, p0.upcase + prest)
187 plural(/#{s0.downcase}(?i)#{srest}$/, p0.downcase + prest)
188 plural(/#{p0.upcase}(?i)#{prest}$/, p0.upcase + prest)
189 plural(/#{p0.downcase}(?i)#{prest}$/, p0.downcase + prest)
190
191 singular(/#{s0.upcase}(?i)#{srest}$/, s0.upcase + srest)
192 singular(/#{s0.downcase}(?i)#{srest}$/, s0.downcase + srest)
193 singular(/#{p0.upcase}(?i)#{prest}$/, s0.upcase + srest)
194 singular(/#{p0.downcase}(?i)#{prest}$/, s0.downcase + srest)
195 end
196 end

207 # Specifies a humanized form of a string by a regular expression rule or
208 # by a string mapping. When using a regular expression based replacement,
209 # the normal humanize formatting is called after the replacement. When a
210 # string is used, the human form should be specified as desired (example:
211 # 'The name', not 'the_name').
212 #
213 # human /_cnt$/i, '\1_count'
214 # human 'legacy_col_person_name', 'Name'

216 @humans.prepend([rule, replacement])
217 end

219 # Clears the loaded inflections within a given scope (default is
220 # <tt>:all</tt>). Give the scope as a symbol of the inflection type, the
221 # options are: <tt>:plurals</tt>, <tt>:singulars</tt>, <tt>:uncountables</tt>,
222 # <tt>:humans</tt>.
223 #
224 # clear :all
225 # clear :plurals
226 def clear(scope = :all)
227 case scope
228 when :all
229 @plurals, @singulars, @uncountables, @humans = [], [], Uncountables.new, []
230 else
231 instance_variable_set "@#{scope}", []
232 end
233 end

237 def define_acronym_regex_patterns
238 @acronym_regex = @acronyms.empty? ? /(?=a)b/ : /#{@acronyms.values.join("|")}/
239 @acronyms_camelize_regex = /^(?:#{@acronym_regex}(?=\b|[A-Z_])|\w)/
240 @acronyms_underscore_regex = /(?:(?<=([A-Za-z\d]))|\b)(#{@acronym_regex})(?=\b|[^a-z])/
241 end
242 end

244 # Yields a singleton instance of Inflector::Inflections so you can specify
245 # additional inflector rules. If passed an optional locale, rules for other
246 # languages can be specified. If not specified, defaults to <tt>:en</tt>.
247 # Only rules for English are provided.
248 #
249 # ActiveSupport::Inflector.inflections(:en) do |inflect|
