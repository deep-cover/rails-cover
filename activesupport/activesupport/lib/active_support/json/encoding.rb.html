40 # escape \u2028 and \u2029 and optionally >, <, & to work around

72 # Convert an object into a "JSON-ready" representation composed of
73 # primitives like Hash, Array, String, Numeric,
74 # and +true+/+false+/+nil+.
75 # Recursively calls #as_json to the object to recursively build a
76 # fully JSON-ready object.
77 #
78 # This allows developers to implement #as_json without having to
79 # worry about what base types of objects they are allowed to return
80 # or having to remember to call #as_json recursively.
81 #
82 # Note: the +options+ hash passed to +object.to_json+ is only passed
83 # to +object.as_json+, not any of this method's recursive +#as_json+
84 # calls.
85 def jsonify(value)
86 case value
87 when String
88 EscapedString.new(value)
89 when Numeric, NilClass, TrueClass, FalseClass
90 value.as_json
91 when Hash
92 Hash[value.map { |k, v| [jsonify(k), jsonify(v)] }]
93 when Array
94 value.map { |v| jsonify(v) }
95 else
96 jsonify value.as_json
97 end
98 end
