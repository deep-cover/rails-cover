30 # Creates a new Store object according to the given options.
31 #
32 # If no arguments are passed to this method, then a new
33 # ActiveSupport::Cache::MemoryStore object will be returned.
34 #
35 # If you pass a Symbol as the first argument, then a corresponding cache
36 # store class under the ActiveSupport::Cache namespace will be created.
37 # For example:
38 #
39 # ActiveSupport::Cache.lookup_store(:memory_store)
40 # # => returns a new ActiveSupport::Cache::MemoryStore object
41 #
42 # ActiveSupport::Cache.lookup_store(:mem_cache_store)
43 # # => returns a new ActiveSupport::Cache::MemCacheStore object
44 #
45 # Any additional arguments will be passed to the corresponding cache store
46 # class's constructor:
47 #
48 # ActiveSupport::Cache.lookup_store(:file_store, '/tmp/cache')
49 # # => same as: ActiveSupport::Cache::FileStore.new('/tmp/cache')
50 #
51 # If the first argument is not a Symbol, then it will simply be returned:
52 #
53 # ActiveSupport::Cache.lookup_store(MyOwnCacheStore.new)
54 # # => returns MyOwnCacheStore.new
55 def lookup_store(*store_option)
56 store, *parameters = *Array.wrap(store_option).flatten
57
58 case store
59 when Symbol
60 retrieve_store_class(store).new(*parameters)
61 when nil
62 ActiveSupport::Cache::MemoryStore.new
63 else
64 store
65 end
66 end

120 # An abstract cache store class. There are multiple cache store
121 # implementations, each having its own additional features. See the classes
122 # under the ActiveSupport::Cache module, e.g.
123 # ActiveSupport::Cache::MemCacheStore. MemCacheStore is currently the most
124 # popular cache store for large production websites.
125 #
126 # Some implementations may not support all methods beyond the basic cache
127 # methods of +fetch+, +write+, +read+, +exist?+, and +delete+.
128 #
129 # ActiveSupport::Cache::Store can store any serializable Ruby object.
130 #
131 # cache = ActiveSupport::Cache::MemoryStore.new
132 #
133 # cache.read('city') # => nil
134 # cache.write('city', "Duckburgh")
135 # cache.read('city') # => "Duckburgh"
136 #
137 # Keys are always translated into Strings and are case sensitive. When an
138 # object is specified as a key and has a +cache_key+ method defined, this
139 # method will be called to define the key. Otherwise, the +to_param+
140 # method will be called. Hashes and Arrays can also be used as keys. The
141 # elements will be delimited by slashes, and the elements within a Hash
142 # will be sorted by key so they are consistent.
143 #
144 # cache.read('city') == cache.read(:city) # => true
145 #
146 # Nil values can be cached.
147 #
148 # If your cache is on a shared infrastructure, you can define a namespace
149 # for your cache entries. If a namespace is defined, it will be prefixed on
150 # to every key. The namespace can be either a static value or a Proc. If it
151 # is a Proc, it will be invoked when each key is evaluated so that you can
152 # use application logic to invalidate keys.
153 #
154 # cache.namespace = -> { @last_mod_time } # Set the namespace to a variable
155 # @last_mod_time = Time.now # Invalidate the entire cache by changing namespace
156 #
157 # Cached data larger than 1kB are compressed by default. To turn off
158 # compression, pass <tt>compress: false</tt> to the initializer or to
159 # individual +fetch+ or +write+ method calls. The 1kB compression
160 # threshold is configurable with the <tt>:compress_threshold</tt> option,
161 # specified in bytes.

189 # Fetches data from the cache, using the given key. If there is data in
190 # the cache with the given key, then that data is returned.
191 #
192 # If there is no such data in the cache (a cache miss), then +nil+ will be
193 # returned. However, if a block has been passed, that block will be passed
194 # the key and executed in the event of a cache miss. The return value of the
195 # block will be written to the cache under the given cache key, and that
196 # return value will be returned.
197 #
198 # cache.write('today', 'Monday')
199 # cache.fetch('today') # => "Monday"
200 #
201 # cache.fetch('city') # => nil
202 # cache.fetch('city') do
203 # 'Duckburgh'
204 # end
205 # cache.fetch('city') # => "Duckburgh"
206 #
207 # You may also specify additional options via the +options+ argument.
208 # Setting <tt>force: true</tt> forces a cache "miss," meaning we treat
209 # the cache value as missing even if it's present. Passing a block is
210 # required when +force+ is true so this always results in a cache write.
211 #
212 # cache.write('today', 'Monday')
213 # cache.fetch('today', force: true) { 'Tuesday' } # => 'Tuesday'
214 # cache.fetch('today', force: true) # => ArgumentError
215 #
216 # The +:force+ option is useful when you're calling some other method to
217 # ask whether you should force a cache write. Otherwise, it's clearer to
218 # just call <tt>Cache#write</tt>.
219 #
220 # Setting <tt>compress: false</tt> disables compression of the cache entry.
221 #
222 # Setting <tt>:expires_in</tt> will set an expiration time on the cache.
223 # All caches support auto-expiring content after a specified number of
224 # seconds. This value can be specified as an option to the constructor
225 # (in which case all entries will be affected), or it can be supplied to
226 # the +fetch+ or +write+ method to effect just one entry.
227 #
228 # cache = ActiveSupport::Cache::MemoryStore.new(expires_in: 5.minutes)
229 # cache.write(key, value, expires_in: 1.minute) # Set a lower value for one entry
230 #
231 # Setting <tt>:version</tt> verifies the cache stored under <tt>name</tt>
232 # is of the same version. nil is returned on mismatches despite contents.
233 # This feature is used to support recyclable cache keys.
234 #
235 # Setting <tt>:race_condition_ttl</tt> is very useful in situations where
236 # a cache entry is used very frequently and is under heavy load. If a
237 # cache expires and due to heavy load several different processes will try
238 # to read data natively and then they all will try to write to cache. To
239 # avoid that case the first process to find an expired cache entry will
240 # bump the cache expiration time by the value set in <tt>:race_condition_ttl</tt>.
241 # Yes, this process is extending the time for a stale value by another few
242 # seconds. Because of extended life of the previous cache, other processes
243 # will continue to use slightly stale data for a just a bit longer. In the
244 # meantime that first process will go ahead and will write into cache the
245 # new value. After that all the processes will start getting the new value.
246 # The key is to keep <tt>:race_condition_ttl</tt> small.
247 #
248 # If the process regenerating the entry errors out, the entry will be
249 # regenerated after the specified number of seconds. Also note that the
250 # life of stale cache is extended only if it expired recently. Otherwise
251 # a new value is generated and <tt>:race_condition_ttl</tt> does not play
252 # any role.
253 #
254 # # Set all values to expire after one minute.
255 # cache = ActiveSupport::Cache::MemoryStore.new(expires_in: 1.minute)
256 #
257 # cache.write('foo', 'original value')
258 # val_1 = nil
259 # val_2 = nil
260 # sleep 60
261 #
262 # Thread.new do
263 # val_1 = cache.fetch('foo', race_condition_ttl: 10.seconds) do
264 # sleep 1
265 # 'new value 1'
266 # end
267 # end
268 #
269 # Thread.new do
270 # val_2 = cache.fetch('foo', race_condition_ttl: 10.seconds) do
271 # 'new value 2'
272 # end
273 # end
274 #
275 # cache.fetch('foo') # => "original value"
276 # sleep 10 # First thread extended the life of cache by another 10 seconds
277 # cache.fetch('foo') # => "new value 1"
278 # val_1 # => "new value 1"
279 # val_2 # => "original value"
280 #
281 # Other options will be handled by the specific cache store implementation.
282 # Internally, #fetch calls #read_entry, and calls #write_entry on a cache
283 # miss. +options+ will be passed to the #read and #write calls.
284 #
285 # For example, MemCacheStore's #write method supports the +:raw+
286 # option, which tells the memcached server to store all values as strings.
287 # We can use this option with #fetch too:
288 #
289 # cache = ActiveSupport::Cache::MemCacheStore.new
290 # cache.fetch("foo", force: true, raw: true) do
291 # :bar
292 # end
293 # cache.fetch('foo') # => "bar"
294 def fetch(name, options = nil)
295 if block_given?
296 options = merged_options(options)
297 key = normalize_key(name, options)
298
299 entry = nil
300 instrument(:read, name, options) do |payload|
301 cached_entry = read_entry(key, options) unless options[:force]
302 entry = handle_expired_entry(cached_entry, key, options)
303 entry = nil if entry && entry.mismatched?(normalize_version(name, options))
304 payload[:super_operation] = :fetch if payload
305 payload[:hit] = !!entry if payload
306 end
307
308 if entry
309 get_entry_value(entry, name, options)
310 else
311 save_block_result_to_cache(name, options) { |_name| yield _name }
312 end
313 elsif options && options[:force]
314 raise ArgumentError, "Missing block: Calling `Cache#fetch` with `force: true` requires a block."
315 else
316 read(name, options)
317 end
318 end

398 # the cache with the given keys, then that data is returned. Otherwise,
399 # the supplied block is called for each key for which there was no data,
400 # and the result will be written to the cache and returned.

419 options = names.extract_options!
420 options = merged_options(options)
421
422 read_multi(*names, options).tap do |results|
423 writes = {}
424
425 (names - results.keys).each do |name|
426 results[name] = writes[name] = yield(name)

624 def handle_expired_entry(entry, key, options)
625 if entry && entry.expired?
626 race_ttl = options[:race_condition_ttl].to_i
627 if (race_ttl > 0) && (Time.now.to_f - entry.expires_at <= race_ttl)
628 # When an entry has a positive :race_condition_ttl defined, put the stale entry back into the cache
629 # for a brief period while the entry is being recalculated.
630 entry.expires_at = Time.now + race_ttl
631 write_entry(key, entry, expires_in: race_ttl * 2)
632 else
633 delete_entry(key, options)
634 end
635 entry = nil
636 end
637 entry
638 end

655 # This class is used to represent cache entries. Cache entries have a value, an optional
656 # expiration time, and an optional version. The expiration time is used to support the :race_condition_ttl option
657 # on the cache. The version is used to support the :version option on the cache for rejecting
658 # mismatches.
659 #
660 # Since cache entries in most instances will be serialized, the internals of this class are highly optimized
661 # using short instance variable names that are lazily defined.

667 # Creates a new cache entry for the specified value. Options supported are
668 # +:compress+, +:compress_threshold+, and +:expires_in+.
669 def initialize(value, options = {})
670 if should_compress?(value, options)
671 @value = compress(value)
672 @compressed = true
673 else
674 @value = value
675 end
676
677 @version = options[:version]
678 @created_at = Time.now.to_f
679 @expires_in = options[:expires_in]
680 @expires_in = @expires_in.to_f if @expires_in
681 end

683 def value
684 compressed? ? uncompress(@value) : @value
685 end
686
687 def mismatched?(version)
688 @version && version && @version != version
689 end

691 # Checks if the entry is expired. The +expires_in+ parameter can override
692 # the value set when the entry was created.
693 def expired?
694 @expires_in && @created_at + @expires_in <= Time.now.to_f
695 end
696
697 def expires_at
698 @expires_in ? @created_at + @expires_in : nil
699 end

702 if value

711 def size

726 # Duplicates the value in a class. This is used by cache implementations that don't natively
727 # serialize entries to protect against accidental cache modifications.
728 def dup_value!
729 if @value && !compressed? && !(@value.is_a?(Numeric) || @value == true || @value == false)
730 if @value.is_a?(String)
731 @value = @value.dup
732 else

735 end

740 if value && options.fetch(:compress, true)
741 compress_threshold = options.fetch(:compress_threshold, DEFAULT_COMPRESS_LIMIT)

747 false
