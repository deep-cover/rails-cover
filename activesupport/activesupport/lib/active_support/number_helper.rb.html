21 # Formats a +number+ into a phone number (US by default e.g., (555)
22 # 123-9876). You can customize the format in the +options+ hash.
23 #
24 # ==== Options
25 #
26 # * <tt>:area_code</tt> - Adds parentheses around the area code.
27 # * <tt>:delimiter</tt> - Specifies the delimiter to use
28 # (defaults to "-").
29 # * <tt>:extension</tt> - Specifies an extension to add to the
30 # end of the generated number.
31 # * <tt>:country_code</tt> - Sets the country code for the phone
32 # number.
33 # * <tt>:pattern</tt> - Specifies how the number is divided into three
34 # groups with the custom regexp to override the default format.
35 # ==== Examples
36 #
37 # number_to_phone(5551234) # => "555-1234"
38 # number_to_phone('5551234') # => "555-1234"
39 # number_to_phone(1235551234) # => "123-555-1234"
40 # number_to_phone(1235551234, area_code: true) # => "(123) 555-1234"
41 # number_to_phone(1235551234, delimiter: ' ') # => "123 555 1234"
42 # number_to_phone(1235551234, area_code: true, extension: 555) # => "(123) 555-1234 x 555"
43 # number_to_phone(1235551234, country_code: 1) # => "+1-123-555-1234"
44 # number_to_phone('123a456') # => "123a456"
45 #
46 # number_to_phone(1235551234, country_code: 1, extension: 1343, delimiter: '.')
47 # # => "+1.123.555.1234 x 1343"
48 #
49 # number_to_phone(75561234567, pattern: /(\d{1,4})(\d{4})(\d{4})$/, area_code: true)
50 # # => "(755) 6123-4567"
51 # number_to_phone(13312345678, pattern: /(\d{3})(\d{4})(\d{4})$/)
52 # # => "133-1234-5678"
53 def number_to_phone(number, options = {})
54 NumberToPhoneConverter.convert(number, options)
55 end

57 # Formats a +number+ into a currency string (e.g., $13.65). You
58 # can customize the format in the +options+ hash.
59 #
60 # The currency unit and number formatting of the current locale will be used
61 # unless otherwise specified in the provided options. No currency conversion
62 # is performed. If the user is given a way to change their locale, they will
63 # also be able to change the relative value of the currency displayed with
64 # this helper. If your application will ever support multiple locales, you
65 # may want to specify a constant <tt>:locale</tt> option or consider
66 # using a library capable of currency conversion.
67 #
68 # ==== Options
69 #
70 # * <tt>:locale</tt> - Sets the locale to be used for formatting
71 # (defaults to current locale).
72 # * <tt>:precision</tt> - Sets the level of precision (defaults
73 # to 2).
74 # * <tt>:unit</tt> - Sets the denomination of the currency
75 # (defaults to "$").
76 # * <tt>:separator</tt> - Sets the separator between the units
77 # (defaults to ".").
78 # * <tt>:delimiter</tt> - Sets the thousands delimiter (defaults
79 # to ",").
80 # * <tt>:format</tt> - Sets the format for non-negative numbers
81 # (defaults to "%u%n"). Fields are <tt>%u</tt> for the
82 # currency, and <tt>%n</tt> for the number.
83 # * <tt>:negative_format</tt> - Sets the format for negative
84 # numbers (defaults to prepending a hyphen to the formatted
85 # number given by <tt>:format</tt>). Accepts the same fields
86 # than <tt>:format</tt>, except <tt>%n</tt> is here the
87 # absolute value of the number.
88 #
89 # ==== Examples
90 #
91 # number_to_currency(1234567890.50) # => "$1,234,567,890.50"
92 # number_to_currency(1234567890.506) # => "$1,234,567,890.51"
93 # number_to_currency(1234567890.506, precision: 3) # => "$1,234,567,890.506"
94 # number_to_currency(1234567890.506, locale: :fr) # => "1 234 567 890,51 €"
95 # number_to_currency('123a456') # => "$123a456"
96 #
97 # number_to_currency(-1234567890.50, negative_format: '(%u%n)')
98 # # => "($1,234,567,890.50)"
99 # number_to_currency(1234567890.50, unit: '&pound;', separator: ',', delimiter: '')
100 # # => "&pound;1234567890,50"
101 # number_to_currency(1234567890.50, unit: '&pound;', separator: ',', delimiter: '', format: '%n %u')
102 # # => "1234567890,50 &pound;"
103 def number_to_currency(number, options = {})
104 NumberToCurrencyConverter.convert(number, options)
105 end

129 # ==== Examples

137 # number_to_percentage(1000, precision: nil) # => "1000%"
138 # number_to_percentage('98a') # => "98a%"
139 # number_to_percentage(100, format: '%n %') # => "100.000 %"

184 # ==== Options

186 # * <tt>:locale</tt> - Sets the locale to be used for formatting

193 # * <tt>:separator</tt> - Sets the separator between the
194 # fractional and integer digits (defaults to ".").

267 # Pretty prints (formats and approximates) a number in a way it
268 # is more readable by humans (eg.: 1200000000 becomes "1.2
269 # Billion"). This is useful for numbers that can get very large
270 # (and too hard to read).
271 #
272 # See <tt>number_to_human_size</tt> if you want to print a file
273 # size.
274 #
275 # You can also define your own unit-quantifier names if you want
276 # to use other decimal units (eg.: 1500 becomes "1.5
277 # kilometers", 0.150 becomes "150 milliliters", etc). You may
278 # define a wide range of unit quantifiers, even fractional ones
279 # (centi, deci, mili, etc).
280 #
281 # ==== Options
282 #
283 # * <tt>:locale</tt> - Sets the locale to be used for formatting
284 # (defaults to current locale).
285 # * <tt>:precision</tt> - Sets the precision of the number
286 # (defaults to 3).
287 # * <tt>:significant</tt> - If +true+, precision will be the number
288 # of significant_digits. If +false+, the number of fractional
289 # digits (defaults to +true+)
290 # * <tt>:separator</tt> - Sets the separator between the
291 # fractional and integer digits (defaults to ".").
292 # * <tt>:delimiter</tt> - Sets the thousands delimiter (defaults
293 # to "").
294 # * <tt>:strip_insignificant_zeros</tt> - If +true+ removes
295 # insignificant zeros after the decimal separator (defaults to
296 # +true+)
297 # * <tt>:units</tt> - A Hash of unit quantifier names. Or a
298 # string containing an i18n scope where to find this hash. It
299 # might have the following keys:
300 # * *integers*: <tt>:unit</tt>, <tt>:ten</tt>,
301 # <tt>:hundred</tt>, <tt>:thousand</tt>, <tt>:million</tt>,
302 # <tt>:billion</tt>, <tt>:trillion</tt>,
303 # <tt>:quadrillion</tt>
304 # * *fractionals*: <tt>:deci</tt>, <tt>:centi</tt>,
305 # <tt>:mili</tt>, <tt>:micro</tt>, <tt>:nano</tt>,
306 # <tt>:pico</tt>, <tt>:femto</tt>
307 # * <tt>:format</tt> - Sets the format of the output string
308 # (defaults to "%n %u"). The field types are:
309 # * %u - The quantifier (ex.: 'thousand')
310 # * %n - The number
311 #
312 # ==== Examples
313 #
314 # number_to_human(123) # => "123"
315 # number_to_human(1234) # => "1.23 Thousand"
316 # number_to_human(12345) # => "12.3 Thousand"
317 # number_to_human(1234567) # => "1.23 Million"
318 # number_to_human(1234567890) # => "1.23 Billion"
319 # number_to_human(1234567890123) # => "1.23 Trillion"
320 # number_to_human(1234567890123456) # => "1.23 Quadrillion"
321 # number_to_human(1234567890123456789) # => "1230 Quadrillion"
322 # number_to_human(489939, precision: 2) # => "490 Thousand"
323 # number_to_human(489939, precision: 4) # => "489.9 Thousand"
324 # number_to_human(1234567, precision: 4,
325 # significant: false) # => "1.2346 Million"
326 # number_to_human(1234567, precision: 1,
327 # separator: ',',
328 # significant: false) # => "1,2 Million"
329 #
330 # number_to_human(500000000, precision: 5) # => "500 Million"
331 # number_to_human(12345012345, significant: false) # => "12.345 Billion"
332 #
333 # Non-significant zeros after the decimal separator are stripped
334 # out by default (set <tt>:strip_insignificant_zeros</tt> to
335 # +false+ to change that):
336 #
337 # number_to_human(12.00001) # => "12"
338 # number_to_human(12.00001, strip_insignificant_zeros: false) # => "12.0"
339 #
340 # ==== Custom Unit Quantifiers
341 #
342 # You can also use your own custom unit quantifiers:
343 # number_to_human(500000, units: { unit: 'ml', thousand: 'lt' }) # => "500 lt"
344 #
345 # If in your I18n locale you have:
346 #
347 # distance:
348 # centi:
349 # one: "centimeter"
350 # other: "centimeters"
351 # unit:
352 # one: "meter"
353 # other: "meters"
354 # thousand:
355 # one: "kilometer"
356 # other: "kilometers"
357 # billion: "gazillion-distance"
358 #
359 # Then you could do:
360 #
361 # number_to_human(543934, units: :distance) # => "544 kilometers"
362 # number_to_human(54393498, units: :distance) # => "54400 kilometers"
363 # number_to_human(54393498000, units: :distance) # => "54.4 gazillion-distance"
364 # number_to_human(343, units: :distance, precision: 1) # => "300 meters"
365 # number_to_human(1, units: :distance) # => "1 meter"
366 # number_to_human(0.34, units: :distance) # => "34 centimeters"
367 def number_to_human(number, options = {})
368 NumberToHumanConverter.convert(number, options)
369 end
