41 def start_element(name, attrs = [])
42 new_hash = { CONTENT_KEY => "".dup }.merge!(Hash[attrs])
43 new_hash[HASH_SIZE_KEY] = new_hash.size + 1
44
45 case current_hash[name]
46 when Array then current_hash[name] << new_hash
47 when Hash then current_hash[name] = [current_hash[name], new_hash]
48 when nil then current_hash[name] = new_hash

52 end

55 if current_hash.length > current_hash.delete(HASH_SIZE_KEY) && current_hash[CONTENT_KEY].blank? || current_hash[CONTENT_KEY] == ""
56 current_hash.delete(CONTENT_KEY)
57 end

59 end
