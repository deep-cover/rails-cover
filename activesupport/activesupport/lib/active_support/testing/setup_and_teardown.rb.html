6module ActiveSupport
7 module Testing
8 # Adds support for +setup+ and +teardown+ callbacks.
9 # These callbacks serve as a replacement to overwriting the
10 # <tt>#setup</tt> and <tt>#teardown</tt> methods of your TestCase.
11 #
12 # class ExampleTest < ActiveSupport::TestCase
13 # setup do
14 # # ...
15 # end
16 #
17 # teardown do
18 # # ...
19 # end
20 # end
21 module SetupAndTeardown
22 extend ActiveSupport::Concern
23
24 included do
25 include ActiveSupport::Callbacks
26 define_callbacks :setup, :teardown
27 end
28
29 module ClassMethods
30 # Add a callback, which runs before <tt>TestCase#setup</tt>.
31 def setup(*args, &block)
32 set_callback(:setup, :before, *args, &block)
33 end
34
35 # Add a callback, which runs after <tt>TestCase#teardown</tt>.
36 def teardown(*args, &block)
37 set_callback(:teardown, :after, *args, &block)
38 end
39 end
40
41 def before_setup # :nodoc:
42 super
43 run_callbacks :setup
44 end
45
46 def after_teardown # :nodoc:
47 run_callbacks :teardown
48 super
49 end
50 end
51 end
