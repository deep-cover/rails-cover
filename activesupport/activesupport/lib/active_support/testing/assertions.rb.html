8 # Asserts that an expression is not truthy. Passes if <tt>object</tt> is
9 # +nil+ or +false+. "Truthy" means "considered true in a conditional"
10 # like <tt>if foo</tt>.
11 #
12 # assert_not nil # => true
13 # assert_not false # => true
14 # assert_not 'foo' # => Expected "foo" to be nil or false
15 #
16 # An error message can be specified.
17 #
18 # assert_not foo, 'foo should be false'
19 def assert_not(object, message = nil)
20 message ||= "Expected #{mu_pp(object)} to be nil or false"
21 assert !object, message
22 end

35 # Test numeric difference between the return value of an expression as a
36 # result of what is evaluated in the yielded block.
37 #
38 # assert_difference 'Article.count' do
39 # post :create, params: { article: {...} }
40 # end
41 #
42 # An arbitrary expression is passed in and evaluated.
43 #
44 # assert_difference 'Article.last.comments(:reload).size' do
45 # post :create, params: { comment: {...} }
46 # end
47 #
48 # An arbitrary positive or negative difference can be specified.
49 # The default is <tt>1</tt>.
50 #
51 # assert_difference 'Article.count', -1 do
52 # post :delete, params: { id: ... }
53 # end
54 #
55 # An array of expressions can also be passed in and evaluated.
56 #
57 # assert_difference [ 'Article.count', 'Post.count' ], 2 do
58 # post :create, params: { article: {...} }
59 # end
60 #
61 # A lambda or a list of lambdas can be passed in and evaluated:
62 #
63 # assert_difference ->{ Article.count }, 2 do
64 # post :create, params: { article: {...} }
65 # end
66 #
67 # assert_difference [->{ Article.count }, ->{ Post.count }], 2 do
68 # post :create, params: { article: {...} }
69 # end
70 #
71 # An error message can be specified.
72 #
73 # assert_difference 'Article.count', -1, 'An Article should be destroyed' do
74 # post :delete, params: { id: ... }
75 # end
76 def assert_difference(expression, difference = 1, message = nil, &block)
77 expressions = Array(expression)
78
79 exps = expressions.map { |e|
80 e.respond_to?(:call) ? e : lambda { eval(e, block.binding) }
81 }
82 before = exps.map(&:call)
83
84 retval = yield
85
86 expressions.zip(exps).each_with_index do |(code, e), i|
87 error = "#{code.inspect} didn't change by #{difference}"
88 error = "#{message}.\n#{error}" if message
89 assert_equal(before[i] + difference, e.call, error)
90 end
91
92 retval
93 end

95 # Assertion that the numeric result of evaluating an expression is not
96 # changed before and after invoking the passed in block.
97 #
98 # assert_no_difference 'Article.count' do
99 # post :create, params: { article: invalid_attributes }
100 # end
101 #

103 #
104 # assert_no_difference 'Article.count', 'An Article should not be created' do
105 # post :create, params: { article: invalid_attributes }
106 # end
107 def assert_no_difference(expression, message = nil, &block)

111 # Assertion that the result of evaluating an expression is changed before
112 # and after invoking the passed in block.
113 #
114 # assert_changes 'Status.all_good?' do
115 # post :create, params: { status: { ok: false } }
116 # end
117 #
118 # You can pass the block as a string to be evaluated in the context of
119 # the block. A lambda can be passed for the block as well.
120 #
121 # assert_changes -> { Status.all_good? } do
122 # post :create, params: { status: { ok: false } }
123 # end
124 #
125 # The assertion is useful to test side effects. The passed block can be
126 # anything that can be converted to string with #to_s.
127 #
128 # assert_changes :@object do
129 # @object = 42
130 # end
131 #
132 # The keyword arguments :from and :to can be given to specify the
133 # expected initial value and the expected value after the block was
134 # executed.
135 #
136 # assert_changes :@object, from: nil, to: :foo do
137 # @object = :foo
138 # end
139 #
140 # An error message can be specified.
141 #
142 # assert_changes -> { Status.all_good? }, 'Expected the status to be bad' do
143 # post :create, params: { status: { incident: true } }
144 # end
145 def assert_changes(expression, message = nil, from: UNTRACKED, to: UNTRACKED, &block)
146 exp = expression.respond_to?(:call) ? expression : -> { eval(expression.to_s, block.binding) }
147
148 before = exp.call
149 retval = yield
150
151 unless from == UNTRACKED
152 error = "#{expression.inspect} isn't #{from.inspect}"
153 error = "#{message}.\n#{error}" if message
154 assert from === before, error
155 end
156
157 after = exp.call
158
159 if to == UNTRACKED
160 error = "#{expression.inspect} didn't change"
161 error = "#{message}.\n#{error}" if message
162 assert before != after, error
163 else
164 error = "#{expression.inspect} didn't change to #{to}"
165 error = "#{message}.\n#{error}" if message
166 assert to === after, error
167 end
168
169 retval
170 end
