52 def deprecate_methods(target_module, *method_names)
53 options = method_names.extract_options!
54 deprecator = options.delete(:deprecator) || self
55 method_names += options.keys
56
57 mod = Module.new do
58 method_names.each do |method_name|
59 define_method(method_name) do |*args, &block|
60 deprecator.deprecation_warning(method_name, options[method_name])
61 super(*args, &block)
62 end
63 end
64 end
65
66 target_module.prepend(mod)
67 end
